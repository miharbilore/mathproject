% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A shipping company has $x$ boxes to transport. If each truck can carry $(x + 5)$ boxes, what is the expression for the total number of boxes transported by $2$ trucks?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x + 10$
    \item $2x - 10$
    \item $x^2 + 10x$
    \item $x^2 - 10x$
    \item $2x^2 + 10x$
  \end{enumerate}

  \item \small The time it takes to travel from City A to City B is given by the expression $t = x^2 - 4x + 4$, where $t$ is time in hours and $x$ is the speed in km/h. What type of trinomial is this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Perfect square trinomial
    \item Difference of cubes
    \item Sum of cubes
    \item Product of two binomials
    \item None of the above
  \end{enumerate}

  \item \small The cost of shipping a package is given by the expression $C = (x + 2)^2$, where $x$ is the weight of the package in kg. What is the expanded form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 2x + 4$
    \item $x^2 - 2x - 4$
    \item $x^2 + 4x + 4$
    \item $x^2 - 4x + 4$
    \item $x^2 + 4x - 4$
  \end{enumerate}

  \item \small The distance between two cities is given by the expression $d = (x - 3)^2$, where $x$ is the speed of the vehicle in km/h. What is the value of $x$ that makes $d = 16$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 1$
    \item $x = 3$
    \item $x = 5$
    \item $x = 7$
    \item $x = 9$
  \end{enumerate}

  \item \small The time it takes for a package to be delivered is given by the expression $t = x^2 - 9$, where $t$ is time in days and $x$ is the distance in km. What is the factored form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x - 3)(x + 3)$
    \item $(x - 3)^2$
    \item $(x + 3)^2$
    \item $(x - 3)(x - 3)$
    \item $(x + 3)(x + 3)$
  \end{enumerate}

  \item \small A logistics company has $x$ packages to deliver. If each delivery truck can carry $(x - 2)$ packages, what is the expression for the total number of packages delivered by $3$ trucks?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3x - 6$
    \item $3x + 6$
    \item $x^2 - 6x$
    \item $x^2 + 6x$
    \item $3x^2 - 6x$
  \end{enumerate}

  \item \small The cost of fuel for a vehicle is given by the expression $C = (x + 1)^2$, where $x$ is the distance traveled in km. What is the expanded form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 2x + 1$
    \item $x^2 - 2x - 1$
    \item $x^2 + 2x - 1$
    \item $x^2 - 2x + 1$
    \item $x^2 + 1$
  \end{enumerate}

  \item \small The time it takes for a vehicle to travel from City A to City B is given by the expression $t = x^2 + 4x + 4$, where $t$ is time in hours and $x$ is the speed in km/h. What type of trinomial is this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Perfect square trinomial
    \item Difference of cubes
    \item Sum of cubes
    \item Product of two binomials
    \item None of the above
  \end{enumerate}

\end{enumerate}
\end{multicols}
