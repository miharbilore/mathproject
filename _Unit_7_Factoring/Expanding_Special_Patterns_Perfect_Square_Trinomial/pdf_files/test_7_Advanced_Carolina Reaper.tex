% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the expansion of $(x + 3)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 6x + 9$
    \item $x^2 + 3x - 9$
    \item $x^2 - 6x + 9$
    \item $x^2 + 9x + 3$
    \item $x^2 - 9x + 3$
  \end{enumerate}

  \item \small Which expression is a perfect square trinomial?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 4x + 4$
    \item $x^2 + 4x - 4$
    \item $x^2 - 4x + 4$
    \item $x^2 + 4x + 5$
    \item $x^2 - 4x - 4$
  \end{enumerate}

  \item \small A bookshelf has $x$ shelves, and each shelf can hold $x + 2$ books. What is the total number of books the bookshelf can hold?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x(x + 2)$
    \item $x(x - 2)$
    \item $x^2 + 2x$
    \item $x^2 - 2x$
    \item $x^2 + 4$
  \end{enumerate}

  \item \small Factor the expression $x^2 + 10x + 25$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 5)^2$
    \item $(x - 5)^2$
    \item $(x + 5)(x - 5)$
    \item $x(x + 10) + 25$
    \item $x(x - 10) - 25$
  \end{enumerate}

  \item \small Tom has been saving money for a new bike and has $x$ dollars in his piggy bank. He wants to buy a bike that costs $x^2 + 5x + 6$ dollars. What is the expansion of the expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 3)(x + 2)$
    \item $(x - 3)(x + 2)$
    \item $(x + 3)(x - 2)$
    \item $x^2 + 5x - 6$
    \item $x^2 - 5x + 6$
  \end{enumerate}

  \item \small A rectangular garden has a length of $x + 3$ meters and a width of $x - 2$ meters. What is the area of the garden?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + x - 6$
    \item $x^2 + 5x + 6$
    \item $x^2 - x - 6$
    \item $x^2 + x + 6$
    \item $x^2 - 5x - 6$
  \end{enumerate}

  \item \small What is the factored form of $x^2 - 7x + 12$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x - 3)(x - 4)$
    \item $(x + 3)(x + 4)$
    \item $(x - 3)(x + 4)$
    \item $(x + 3)(x - 4)$
    \item $x(x - 7) + 12$
  \end{enumerate}

  \item \small Ashley is making a quilt with $x$ rows of squares, and each row has $x + 2$ squares. What is the total number of squares in the quilt?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x(x + 2)$
    \item $x(x - 2)$
    \item $x^2 + 2x$
    \item $x^2 - 2x$
    \item $x^2 + 4$
  \end{enumerate}

\end{enumerate}
\end{multicols}
