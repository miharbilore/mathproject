% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the expansion of $(x + 2)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 4x + 4$
    \item $x^2 + 2x + 4$
    \item $x^2 - 4x + 4$
    \item $x^2 + 4x - 4$
    \item $x^2 - 2x - 4$
  \end{enumerate}

  \item \small A hiker is traveling at a rate of $x$ miles per hour. If the hiker's speed is increased by 3 miles per hour, what is the expansion of $(x + 3)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 6x + 9$
    \item $x^2 + 3x + 9$
    \item $x^2 - 6x + 9$
    \item $x^2 + 6x - 9$
    \item $x^2 - 3x - 9$
  \end{enumerate}

  \item \small What is the factored form of the perfect square trinomial $x^2 + 10x + 25$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 5)^2$
    \item $(x - 5)^2$
    \item $(x + 2)^2$
    \item $(x - 2)^2$
    \item $(x + 3)^2$
  \end{enumerate}

  \item \small A park ranger is tracking the population of a species of animal. If the population is currently $x$ and is increasing at a rate of 4 animals per year, what is the expansion of $(x + 4)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 8x + 16$
    \item $x^2 + 4x + 16$
    \item $x^2 - 8x + 16$
    \item $x^2 + 8x - 16$
    \item $x^2 - 4x - 16$
  \end{enumerate}

  \item \small What is the expansion of $(2x - 3)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4x^2 - 12x + 9$
    \item $4x^2 + 12x + 9$
    \item $4x^2 - 12x - 9$
    \item $4x^2 + 12x - 9$
    \item $4x^2 + 6x + 9$
  \end{enumerate}

  \item \small A group of friends are planning a road trip. If their car gets $x$ miles per gallon and they drive for 2 hours at a speed of $x$ miles per hour, what is the expansion of $(x cdot 2)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4x^2$
    \item $2x^2$
    \item $x^2$
    \item $4x$
    \item $2x$
  \end{enumerate}

  \item \small What is the factored form of the perfect square trinomial $4x^2 + 12x + 9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(2x + 3)^2$
    \item $(2x - 3)^2$
    \item $(x + 3)^2$
    \item $(x - 3)^2$
    \item $(x + 2)^2$
  \end{enumerate}

  \item \small A farmer is harvesting crops at a rate of $x$ bushels per hour. If the farmer's harvesting speed is increased by 2 bushels per hour, what is the expansion of $(x + 2)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 4x + 4$
    \item $x^2 + 2x + 4$
    \item $x^2 - 4x + 4$
    \item $x^2 + 4x - 4$
    \item $x^2 - 2x - 4$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A park is planning to build a new hiking trail. The trail will be $x$ miles long and will have a total elevation gain of $x^2 + 6x + 9$ feet. Factor the expression $x^2 + 6x + 9$ and explain its significance in the context of the hiking trail. \vspace{2cm}

  \item \small A group of friends are planning a road trip to a national park. Their car gets $x$ miles per gallon, and they expect to drive for $x + 4$ hours. What is the expansion of $(x + 4)^2$? Use this expansion to calculate the total distance they will travel, assuming they drive at a constant speed of $x$ miles per hour. \vspace{2cm}

\end{enumerate}

\begin{tcolorbox}[colback=blue!5, colframe=mainblue!50, title=\small \textbf{Chef's Tips}, fonttitle=\bfseries, bottom=1mm]
\begin{itemize}[noitemsep, topsep=2pt, leftmargin=0.5cm]
  \item \scriptsize Tip 1: Encourage students to use the formula $(a + b)^2 = a^2 + 2ab + b^2$ to expand special patterns.\item \scriptsize Tip 2: Remind students to look for perfect square trinomials in the form $a^2 + 2ab + b^2$ and factor them as $(a + b)^2$.\item \scriptsize Tip 3: Emphasize the importance of checking their work by plugging their factored expressions back into the original equation.
\end{itemize}
