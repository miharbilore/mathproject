% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the expansion of $(x + 5)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 10x + 25$
    \item $x^2 - 10x + 25$
    \item $x^2 + 5x + 10$
    \item $x^2 - 5x - 10$
    \item $x^2 + 25$
  \end{enumerate}

  \item \small If $y = 3x^2 + 12x + 12$, is this a perfect square trinomial?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Yes
    \item No
    \item Maybe
    \item Insufficient information
    \item Only for certain values of $x$
  \end{enumerate}

  \item \small What is the factorization of $x^2 - 14x + 49$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x - 7)^2$
    \item $(x + 7)^2$
    \item $(x - 7)(x + 7)$
    \item $x(x - 14) + 49$
    \item $x(x + 14) - 49$
  \end{enumerate}

  \item \small The temperature of a city can be modeled by $T(x) = x^2 - 4x + 4$, where $x$ is the number of hours after sunrise. What is the factored form of $T(x)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x - 2)^2$
    \item $(x + 2)^2$
    \item $x(x - 4)$
    \item $(x - 4)(x + 1)$
    \item (x + 2)(x - 2)
  \end{enumerate}

  \item \small A weather balloons' altitude can be represented as $h = t^2 - 6t + 8$. Is the expression $h$ a perfect square trinomial?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Yes
    \item No
    \item For $t > 0$
    \item Only if $t$ is a real number
    \item Not enough information
  \end{enumerate}

  \item \small The area of a rectangular garden is $A = x^2 + 10x + 25$. What is the length of the sides of this garden?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5m and 5m
    \item 10m and 10m
    \item 5m and 10m
    \item $x$ m and $(x + 5)$m
    \item $(x + 5)$ m and $(x + 5)$ m
  \end{enumerate}

  \item \small Water level of a river can be represented as $W(x) = x^2 - 8x + 16$, where $x$ is the distance in kilometers from the dam. What is $W(4)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $4$
    \item $8$
    \item $16$
    \item $32$
  \end{enumerate}

  \item \small In an ecosystem, population of a certain species is given by $P(x) = x^2 + 2x + 1$. Factorize the population equation.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 1)^2$
    \item $(x - 1)^2$
    \item $(x + 2)(x + 1)$
    \item $(x - 2)(x - 1)$
    \item $(x + 1)(x - 1)$
  \end{enumerate}

\end{enumerate}
\end{multicols}
