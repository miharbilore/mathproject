\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small The ancient Egyptians built pyramids using $a^2 + 2ab + b^2$ blocks. What is the factored form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(a+b)^2$
    \item $(a-b)^2$
    \item $(ab)^2$
    \item $a^2 + b^2$
    \item $2ab$
  \end{enumerate}

  \item \small The Babylonians used the formula $x^2 - 2xy + y^2$ to calculate areas. What is the factored form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x+y)^2$
    \item $(x-y)^2$
    \item $(xy)^2$
    \item $x^2 + y^2$
    \item $2xy$
  \end{enumerate}

  \item \small The ancient Greeks expanded the expression $(x+3)^2$. What is the result?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 6x + 9$
    \item $x^2 - 6x + 9$
    \item $x^2 + 3x + 9$
    \item $x^2 - 3x + 9$
    \item $x^2 + 9$
  \end{enumerate}

  \item \small The Mayans used the expression $a^2 - 2ab + b^2$ to calculate the area of a rectangle. What is the factored form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(a+b)^2$
    \item $(a-b)^2$
    \item $(ab)^2$
    \item $a^2 + b^2$
    \item $2ab$
  \end{enumerate}

  \item \small The Chinese expanded the expression $(x-2)^2$. What is the result?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 4x + 4$
    \item $x^2 - 4x + 4$
    \item $x^2 + 2x + 4$
    \item $x^2 - 2x + 4$
    \item $x^2 + 4$
  \end{enumerate}

  \item \small The ancient Indians used the formula $x^2 + 2xy + y^2$ to calculate volumes. What is the factored form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x+y)^2$
    \item $(x-y)^2$
    \item $(xy)^2$
    \item $x^2 + y^2$
    \item $2xy$
  \end{enumerate}

  \item \small The Phoenicians expanded the expression $(x+5)^2$. What is the result?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 10x + 25$
    \item $x^2 - 10x + 25$
    \item $x^2 + 5x + 25$
    \item $x^2 - 5x + 25$
    \item $x^2 + 25$
  \end{enumerate}

  \item \small The ancient Sumerians used the expression $a^2 + 2ab + b^2$ to calculate areas. What is the factored form of this expression?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(a+b)^2$
    \item $(a-b)^2$
    \item $(ab)^2$
    \item $a^2 + b^2$
    \item $2ab$
  \end{enumerate}
