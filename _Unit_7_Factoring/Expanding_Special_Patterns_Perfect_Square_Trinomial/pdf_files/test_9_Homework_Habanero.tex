% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the expansion of $(a + b)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $a^2 + 2ab + b^2$
    \item $a^2 - 2ab + b^2$
    \item $a^2 + ab + b^2$
    \item $a^2 - ab + b^2$
    \item $a^2 + b^2$
  \end{enumerate}

  \item \small A doctor prescribes a dosage of medicine that is $2x + 5$ mg, where $x$ is the number of days. If the patient takes the medicine for $3$ days, what is the total dosage in mg?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$ mg
    \item $21$ mg
    \item $23$ mg
    \item $25$ mg
    \item $30$ mg
  \end{enumerate}

  \item \small What is the factored form of $x^2 + 10x + 25$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 5)^2$
    \item $(x - 5)^2$
    \item $(x + 10)^2$
    \item $(x - 10)^2$
    \item $x(x + 25)$
  \end{enumerate}

  \item \small The growth rate of a cell culture is modeled by the equation $P = 2t^2 + 12t + 10$, where $P$ is the population size and $t$ is time in hours. What is the population size after $2$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $42$
    \item $50$
    \item $52$
    \item $54$
    \item $60$
  \end{enumerate}

  \item \small What is the expanded form of $(a - b)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $a^2 + 2ab + b^2$
    \item $a^2 - 2ab + b^2$
    \item $a^2 + ab + b^2$
    \item $a^2 - ab + b^2$
    \item $a^2 + b^2$
  \end{enumerate}

  \item \small A medical researcher collects data on the number of patients who respond to a new treatment. The results are modeled by the equation $y = x^2 - 4x + 4$, where $y$ is the number of patients and $x$ is the dosage level. What is the value of $y$ when $x = 2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $2$
    \item $4$
    \item $6$
    \item $8$
  \end{enumerate}

  \item \small The concentration of a solution is given by the equation $C = 2x^2 + 8x + 6$, where $C$ is the concentration and $x$ is the amount of solute. What is the concentration when $x = 1$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$
    \item $14$
    \item $16$
    \item $18$
    \item $20$
  \end{enumerate}

  \item \small What is the factored form of $x^2 - 6x + 9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 3)^2$
    \item $(x - 3)^2$
    \item $(x + 6)^2$
    \item $(x - 6)^2$
    \item $x(x - 9)$
  \end{enumerate}

\end{enumerate}
