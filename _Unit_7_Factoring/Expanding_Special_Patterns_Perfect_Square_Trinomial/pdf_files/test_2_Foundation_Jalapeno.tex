% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the expansion of $(x + 3)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 6x + 9$
    \item $x^2 - 6x + 9$
    \item $x^2 + 3x - 9$
    \item $x^2 - 3x - 9$
    \item $x^2 + 9x + 3$
  \end{enumerate}

  \item \small Which of the following is a perfect square trinomial?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 4x + 4$
    \item $x^2 - 4x - 4$
    \item $x^2 + 2x - 4$
    \item $x^2 - 2x + 4$
    \item $x^2 + 4x - 2$
  \end{enumerate}

  \item \small What is the factored form of $x^2 + 10x + 25$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x + 5)^2$
    \item $(x - 5)^2$
    \item $(x + 2)(x + 8)$
    \item $(x - 2)(x - 8)$
    \item $(x + 3)(x + 7)$
  \end{enumerate}

  \item \small The gravitational force between two planets can be modeled by $F = \frac{Gm_1m_2}{r^2}$, where $m_1$ and $m_2$ are the masses of the planets and $r$ is the distance between them. What is the expansion of $(r + 2)^2$ in this context?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $r^2 + 4r + 4$
    \item $r^2 - 4r + 4$
    \item $r^2 + 2r - 4$
    \item $r^2 - 2r + 4$
    \item $r^2 + 4r - 2$
  \end{enumerate}

  \item \small A lab experiment involves the expansion of $(x - 4)^2$. What is the result?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 - 8x + 16$
    \item $x^2 + 8x + 16$
    \item $x^2 - 4x - 16$
    \item $x^2 + 4x - 16$
    \item $x^2 - 2x + 8$
  \end{enumerate}

  \item \small A spacecraft's trajectory can be modeled using the equation $d^2 = (v_0t + \frac{1}{2}at^2)^2$. What is the expansion of $(v_0t + \frac{1}{2}at^2)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $v_0^2t^2 + v_0at^3 + \frac{1}{4}a^2t^4$
    \item $v_0^2t^2 - v_0at^3 + \frac{1}{4}a^2t^4$
    \item $v_0^2t^2 + 2v_0at^3 + a^2t^4$
    \item $v_0^2t^2 - 2v_0at^3 + a^2t^4$
    \item $v_0^2t^2 + v_0at^3 - \frac{1}{4}a^2t^4$
  \end{enumerate}

  \item \small The expansion of $(x + 2y)^2$ is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 4xy + 4y^2$
    \item $x^2 - 4xy + 4y^2$
    \item $x^2 + 2xy - 4y^2$
    \item $x^2 - 2xy - 4y^2$
    \item $x^2 + 4xy - 2y^2$
  \end{enumerate}

  \item \small Which of the following is not a perfect square trinomial?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + 6x + 9$
    \item $x^2 - 6x + 9$
    \item $x^2 + 2x - 4$
    \item $x^2 - 2x + 4$
    \item $x^2 + 4x + 4$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small The distance between two galaxies can be modeled by $d = (v_0t + \frac{1}{2}at^2)$. Expand $(v_0t + \frac{1}{2}at^2)^2$ and show your work. What is the significance of the expansion in this context? \vspace{2cm}

  \item \small A satellite's position can be determined using the equation $x^2 + 10x + 25 = (x + 5)^2$. Factor the left side of the equation and compare it to the right side. What does this tell you about the satellite's position? Use math to support your answer. \vspace{2cm}

\end{enumerate}
