% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the expansion of $(a+b)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $a^2 + 2ab + b^2$
    \item $a^2 - 2ab + b^2$
    \item $a^2 + ab + b^2$
    \item $a^2 - ab + b^2$
    \item $a^2 + b^2$
  \end{enumerate}

  \item \small What is the factored form of $x^2 + 12x + 36$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x+6)^2$
    \item $(x-6)^2$
    \item $(x+6)(x-6)$
    \item $(x+3)(x+12)$
    \item $(x-3)(x-12)$
  \end{enumerate}

  \item \small A basketball player scored $x$ points in the first half and $x+12$ points in the second half. What is the expansion of $(x+x+12)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x+12$
    \item $2x-12$
    \item $x+12$
    \item $x-12$
    \item $x^2+12$
  \end{enumerate}

  \item \small What is the value of $a$ in the perfect square trinomial $a^2 + 20a + 100$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 20
    \item 5
    \item 4
    \item 1
  \end{enumerate}

  \item \small A soccer player kicks the ball $x$ meters in the first half and $(x+5)$ meters in the second half. What is the expansion of $(x+x+5)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x+5$
    \item $2x-5$
    \item $x+5$
    \item $x-5$
    \item $x^2+5$
  \end{enumerate}

  \item \small What is the factored form of $x^2 - 16x + 64$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(x-8)^2$
    \item $(x+8)^2$
    \item $(x-8)(x+8)$
    \item $(x-4)(x-16)$
    \item $(x+4)(x+16)$
  \end{enumerate}

  \item \small A cyclist rides $x$ kilometers in the first hour and $(x-4)$ kilometers in the second hour. What is the expansion of $(x+x-4)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x-4$
    \item $2x+4$
    \item $x-4$
    \item $x+4$
    \item $x^2-4$
  \end{enumerate}

  \item \small What is the expansion of $(x-5)^2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 - 10x + 25$
    \item $x^2 + 10x + 25$
    \item $x^2 - 10x - 25$
    \item $x^2 + 10x - 25$
    \item $x^2 + 25$
  \end{enumerate}

\end{enumerate}
\end{multicols}
