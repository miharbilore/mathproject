  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]

    \item 4y
    \item 0
    \item 1
  \end{enumerate}

  \item \small What is the operator in the expression $3 + 2$?

    \item 1

    \item $+$

  \item \small What does the term $3x$ represent?
