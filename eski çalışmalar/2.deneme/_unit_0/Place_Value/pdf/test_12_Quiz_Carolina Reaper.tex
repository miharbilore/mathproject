\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A cargo ship can transport $432$ containers. What is the value of the digit $4$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $400$
    \item $40$
    \item $4$
    \item $432$
    \item $43$
  \end{enumerate}

  \item \small Rounding $247$ to the nearest ten gives
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $200$
    \item $250$
    \item $240$
    \item $230$
    \item $260$
  \end{enumerate}

  \item \small Write the number $945$ in expanded form as $900 + 40 + 5$. What is the value of the digit $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $900$
    \item $90$
    \item $9$
    \item $45$
    \item $945$
  \end{enumerate}

  \item \small A logistics company has $753$ packages to deliver. The digit $7$ represents
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item hundreds
    \item tens
    \item ones
    \item thousands
    \item ten thousands
  \end{enumerate}

  \item \small The number $6,542$ rounded to the nearest thousand is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6,000$
    \item $7,000$
    \item $5,000$
    \item $8,000$
    \item $9,000$
  \end{enumerate}

  \item \small A truck driver travels $945$ miles. The digit $9$ represents
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item tens
    \item ones
    \item hundreds
    \item thousands
    \item ten thousands
  \end{enumerate}

  \item \small Write the number $753$ in standard form as $7 \cdot 10^2 + 5 \cdot 10^1 + 3 \cdot 10^0$. What is the value of the digit $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $50$
    \item $5$
    \item $500$
    \item $753$
    \item $7$
  \end{enumerate}

  \item \small A shipping company has $2,943$ boxes to transport. Rounding this number to the nearest hundred gives
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2,900$
    \item $3,000$
    \item $2,940$
    \item $3,100$
    \item $2,950$
  \end{enumerate}

\end{enumerate}
