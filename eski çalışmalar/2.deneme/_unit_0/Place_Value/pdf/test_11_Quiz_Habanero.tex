% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small In the number $456$, what is the value of the digit $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ hundreds
    \item $5$ tens
    \item $5$ ones
    \item $5$ thousands
    \item $5$ ten-thousands
  \end{enumerate}

  \item \small The average temperature in a region is $24.67$ degrees. What is this number rounded to the nearest tenth?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $24.7$
    \item $24.6$
    \item $25.0$
    \item $24.0$
    \item $25.7$
  \end{enumerate}

  \item \small Write the number $945$ in expanded form.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $900 + 40 + 5$
    \item $900 + 50 + 5$
    \item $900 + 40 + 4$
    \item $900 + 40 + 6$
    \item $900 + 50 + 4$
  \end{enumerate}

  \item \small A weather station recorded $117.3$ mm of rainfall. What is the value of the digit $1$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$ hundred
    \item $1$ ten
    \item $1$ one
    \item $1$ tenth
    \item $1$ hundredth
  \end{enumerate}

  \item \small The ecosystem of a forest is balanced when the temperature is between $18$ and $22$ degrees. What is the range of possible temperatures in this ecosystem?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $18 < x < 22$
    \item $18 \leq x \leq 22$
    \item $18 < x \leq 22$
    \item $18 \leq x < 22$
    \item $18 = x = 22$
  \end{enumerate}

  \item \small A scientist measured the temperature of a lake to be $15.23$ degrees. What is this number in standard form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15.23$
    \item $1.523 \times 10^{1}$
    \item $1.523 \times 10^{0}$
    \item $1.523 \times 10^{-1}$
    \item $1.523 \times 10^{-2}$
  \end{enumerate}

  \item \small The water level in a reservoir rose by $23.45$ meters. What is the value of the digit $4$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4$ tens
    \item $4$ ones
    \item $4$ tenths
    \item $4$ hundredths
    \item $4$ thousands
  \end{enumerate}

  \item \small A city's average annual rainfall is $876.25$ mm. What is this number rounded to the nearest hundred?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $900$
    \item $800$
    \item $870$
    \item $880$
    \item $875$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\end{enumerate}
