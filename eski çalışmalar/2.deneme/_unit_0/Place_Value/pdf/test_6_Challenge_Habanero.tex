\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small In a gaming scoreboard, a player's score is $4321$. What is the value of the digit $4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Ones place
    \item Tens place
    \item Hundreds place
    \item Thousands place
    \item Ten thousands place
  \end{enumerate}

  \item \small A coder used $658$ MB of data. Rounded to the nearest ten, how much data was used?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 650
    \item 660
    \item 700
    \item 600
    \item 500
  \end{enumerate}

  \item \small In a coding logic problem, the number $945$ is given. What is the value of the digit $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Ones place
    \item Tens place
    \item Hundreds place
    \item Thousands place
    \item None of the above
  \end{enumerate}

  \item \small A gamer's score is $1200$. What is the expanded form of this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1000 + 200$
    \item $1000 + 200 + 0$
    \item $1000 + 20 + 0$
    \item $1000 + 2 + 0$
    \item $100 + 20 + 0$
  \end{enumerate}

  \item \small A coding challenge requires a data usage of $753$ GB. Rounded to the nearest hundred, how much data is required?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 700
    \item 800
    \item 750
    \item 500
    \item 600
  \end{enumerate}

  \item \small In a tech review, a smartphone's storage is $8192$ MB. What is the standard form of this number?

    \item $8 \times 10^2$
    \item $8 \times 10^1$
    \item $8 \times 10^4$
    \item $8 \times 10^0$
  \end{enumerate}

  \item \small A programmer used $279$ hours of coding time. Rounded to the nearest ten, how many hours were used?

    \item 270
    \item 290
    \item 300
    \item 200
  \end{enumerate}

    \item Hundreds place
    \item Thousands place
