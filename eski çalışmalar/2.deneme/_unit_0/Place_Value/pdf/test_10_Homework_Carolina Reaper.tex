% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A rhythm pattern has $4$ beats in a measure, and there are $32$ measures in a song. What is the total number of beats in the song in standard form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $40$
    \item $80$
    \item $100$
    \item $128$
    \item $160$
  \end{enumerate}

  \item \small An art canvas is $48$ inches wide. What is the value of the digit $4$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Tens place
    \item Ones place
    \item Hundreds place
    \item Tens place
    \item Forties place
  \end{enumerate}

  \item \small A music frequency is measured at $432$ Hz. What is this number in expanded form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $400 + 30 + 2$
    \item $300 + 100 + 30 + 2$
    \item $400 + 30 + 2$
    \item $400 + 20 + 12$
    \item $300 + 100 + 32$
  \end{enumerate}

  \item \small A painting is $123.45$ meters long. Rounded to the nearest tenth, what is its length?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $120.0$
    \item $120.5$
    \item $123.5$
    \item $123.4$
    \item $124.0$
  \end{enumerate}

  \item \small There are $256$ seats in a concert hall. What is the value of the digit $5$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Ones place
    \item Tens place
    \item Hundreds place
    \item No value
    \item Thousands place
  \end{enumerate}

  \item \small A song is $4.23$ minutes long. What is $4.23$ in expanded form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4 + rac{2}{10} + rac{3}{100}$
    \item $4 + rac{23}{100}$
    \item $4 + rac{2}{10} + rac{3}{1000}$
    \item $rac{4}{1} + rac{2}{10} + rac{3}{100}$
    \item $rac{423}{100}$
  \end{enumerate}

  \item \small A painting is $75.6$ cm wide. Rounded to the nearest whole number, what is its width?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $70$
    \item $71$
    \item $75$
    \item $76$
    \item $80$
  \end{enumerate}

  \item \small A music album has $421$ songs. What is the value of the digit $2$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Ones place
    \item Tens place
    \item Hundreds place
    \item No value
    \item Thousands place
  \end{enumerate}

\end{enumerate}
