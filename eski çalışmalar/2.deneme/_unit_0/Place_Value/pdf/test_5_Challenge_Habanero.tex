\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the value of the digit 5 in the number $456,789$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5,000$
    \item $500$
    \item $50$
    \item $5$

  \end{enumerate}

  \item \small A hiker travels $234.56$ miles in a day. Rounded to the nearest ten, how many miles did the hiker travel?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $200$
    \item $230$
    \item $240$
    \item $250$
    \item $300$
  \end{enumerate}

  \item \small A wildlife reserve has $14,567$ animals. Write this number in expanded form.

    \item $10,000 + 4,000 + 500 + 60 + 7$
    \item $10,000 + 4,000 + 500 + 60 + 6$
    \item $10,000 + 4,000 + 500 + 70 + 7$
    \item $10,000 + 4,000 + 600 + 60 + 7$
    \item $10,000 + 4,000 + 500 + 50 + 17$
  \end{enumerate}

  \item \small What is the standard form of the number $2 \cdot 10^3 + 5 \cdot 10^2 + 6 \cdot 10^1 + 8 \cdot 10^0$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2,568$
    \item $2,658$
    \item $2,586$
    \item $2,056$
    \item $2,065$
  \end{enumerate}

  \item \small A park ranger measures the length of a trail to be $4,321.9$ meters. Rounded to the nearest hundred, what is the length of the trail?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4,300$
    \item $4,400$
    \item $4,200$
    \item $4,500$
    \item $4,000$
  \end{enumerate}

  \item \small What is the value of the digit 9 in the number $987,654$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $900,000$
    \item $90,000$
    \item $9,000$
    \item $900$
    \item $90$
  \end{enumerate}

  \item \small A group of friends plan to drive $278.5$ miles to a campsite. If their car gets $25$ miles per gallon, how many gallons of fuel will they use, rounded to the nearest tenth?

    \item $11.1$
    \item $12.3$
    \item $11.2$
