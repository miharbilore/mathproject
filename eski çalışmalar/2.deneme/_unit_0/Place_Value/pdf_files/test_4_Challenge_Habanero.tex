% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small In a recipe for $45 \frac{3}{4}$ cups of flour, what is the value of the digit 4?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Thousands
    \item Hundreds
    \item Tens
    \item Ones
    \item Tenths
  \end{enumerate}

  \item \small A building is $234.56$ meters tall. Rounded to the nearest hundred, what is its height?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $200$
    \item $300$
    \item $234$
    \item $235$
    \item $240$
  \end{enumerate}

  \item \small The cost of $3 \frac{1}{2}$ acres of land is $12,000$. What is the cost per acre in standard form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3,428.57$
    \item $3,571.43$
    \item $3,600$
    \item $3,429$
    \item $3,571$
  \end{enumerate}

  \item \small A chef needs $15 \frac{3}{4}$ kilograms of sugar. What is this in expanded form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15 + \frac{3}{4}$
    \item $15 + 0.75$
    \item $15 \cdot \frac{3}{4}$
    \item $15 \cdot 0.75$
    \item $15 + 3 \cdot \frac{1}{4}$
  \end{enumerate}

  \item \small The length of a room is $12.34$ meters. Rounded to the nearest tenth, what is its length?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12.3$
    \item $12.4$
    \item $12$
    \item $12.34$
    \item $12.5$
  \end{enumerate}

  \item \small A recipe calls for $0.45$ liters of oil. What is this in standard form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $rac{45}{100}$
    \item $rac{9}{20}$
    \item $45$
    \item $rac{45}{10}$
    \item $rac{4.5}{10}$
  \end{enumerate}

  \item \small A water tank can hold $456.78$ liters. What is the value of the digit 6?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Ones
    \item Tens
    \item Hundreds
    \item Thousands
    \item Tenths
  \end{enumerate}

  \item \small The cost of a meal is $23.49$. Rounded to the nearest dollar, what is the cost?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $23$
    \item $24$
    \item $25$
    \item $30$
  \end{enumerate}

\end{enumerate}
