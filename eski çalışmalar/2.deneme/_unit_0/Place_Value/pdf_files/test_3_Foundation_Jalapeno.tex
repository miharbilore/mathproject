% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small In the number $425$, what is the value of the digit $2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $200$
    \item $20$
    \item $2$
    \item $25$
    \item $250$
  \end{enumerate}

  \item \small A marathon runner covers a distance of $27.83$ kilometers. Rounded to the nearest tenth, the distance is $=$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $27.8$
    \item $27.9$
    \item $28.0$
    \item $28.3$
    \item $27.7$
  \end{enumerate}

  \item \small The score of a basketball game is $94$ to $67$. What is the value of the digit $9$ in $94$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $9$
    \item $90$
    \item $94$
    \item $67$
    \item $100$
  \end{enumerate}

  \item \small A cyclist travels $14.29$ miles. In expanded form, this is $=$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10 + 4 + 0.2 + 0.09$
    \item $10 + 4 + 0.2 + 0.09 + 0.001$
    \item $10 + 4 + 0.20 + 0.09$
    \item $10 + 4 + 0.29$
    \item $14 + 0.29$
  \end{enumerate}

  \item \small A sports statistician records $43,219$ attendance. What is the value of the digit $1$ in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $10$
    \item $100$
    \item $1,000$
    \item $10,000$
  \end{enumerate}

  \item \small Rounded to the nearest hundred, $749$ is $=$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $700$
    \item $750$
    \item $800$
    \item $7000$
    \item $7500$
  \end{enumerate}

  \item \small A sprinter runs $9.45$ seconds. In standard form, the time is $=$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $9 + 0.4 + 0.05$
    \item $9 + 0.45$
    \item $9.45$
    \item $945$
    \item $94.5$
  \end{enumerate}

  \item \small In a tennis match, a player wins $6$ sets out of $8$. If each set is worth $100$ points, what is the total value of the sets won, rounded to the nearest hundred?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $500$
    \item $600$
    \item $700$
    \item $6000$
    \item $800$
  \end{enumerate}

\end{enumerate}
\end{multicols}
