% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the value of the digit 5 in the number $5\cdot10^3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5\cdot10^2$
    \item $5\cdot10^1$
    \item $5\cdot10^0$
    \item $5\cdot10^3$
    \item $5\cdot10^4$
  \end{enumerate}

  \item \small Round $4,256$ to the nearest thousand
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4,000$
    \item $5,000$
    \item $3,000$
    \item $0$
    \item $10,000$
  \end{enumerate}

  \item \small What is $7,492$ in expanded form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $7\cdot10^3+4\cdot10^2+9\cdot10^1+2\cdot10^0$
    \item $7\cdot10^0+4\cdot10^1+9\cdot10^2+2\cdot10^3$
    \item $7\cdot10^2+4\cdot10^1+9\cdot10^0+2\cdot10^3$
    \item $7\cdot10^3+4\cdot10^2+9\cdot10^1+2\cdot10^0$
    \item $7\cdot10^1+4\cdot10^2+9\cdot10^3+2\cdot10^0$
  \end{enumerate}

  \item \small The CEO of a company has $25,934$ shares of stock. What is the value of the digit 9?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $9\cdot10^0$
    \item $9\cdot10^1$
    \item $9\cdot10^2$
    \item $9\cdot10^3$
    \item $9\cdot10^4$
  \end{enumerate}

  \item \small A savings account has $1,234.56$ dollars. Round this number to the nearest ten
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1,230$
    \item $1,240$
    \item $1,200$
    \item $1,235$
    \item $1,250$
  \end{enumerate}

  \item \small What is the standard form of $3\cdot10^2+4\cdot10^1+2\cdot10^0$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $324$
    \item $342$
    \item $322$
    \item $324.2$
    \item $340$
  \end{enumerate}

  \item \small The tax on a $10,943$ dollar investment is $10\%$. What is the tax amount rounded to the nearest dollar?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1,094$
    \item $1,000$
    \item $1,095$
    \item $1,094.30$
    \item $109$
  \end{enumerate}

  \item \small The value of a stock increases by $5,678$ dollars. What is the value of the digit 6 in this number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6\cdot10^3$
    \item $6\cdot10^1$
    \item $6\cdot10^2$
    \item $6\cdot10^0$
    \item $6\cdot10^4$
  \end{enumerate}

\end{enumerate}
\end{multicols}
