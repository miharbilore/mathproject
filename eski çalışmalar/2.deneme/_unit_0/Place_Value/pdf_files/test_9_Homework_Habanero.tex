% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the value of the digit $7$ in the number $74521$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $7 \cdot 10^4$
    \item $7 \cdot 10^3$
    \item $7 \cdot 10^2$
    \item $7 \cdot 10^1$
    \item $7 \cdot 10^0$
  \end{enumerate}

  \item \small Round $456.789$ to the nearest $10$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $450$
    \item $460$
    \item $457$
    \item $456.8$
    \item $456.79$
  \end{enumerate}

  \item \small A patient needs $0.0456$ liters of medicine. Write this in expanded form
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4 \cdot 10^{-2} + 5 \cdot 10^{-3} + 6 \cdot 10^{-4}$
    \item $4 \cdot 10^{-1} + 5 \cdot 10^{-2} + 6 \cdot 10^{-3}$
    \item $4 \cdot 10^{-2} + 5 \cdot 10^{-3} + 6 \cdot 10^{-4}$
    \item $45 \cdot 10^{-3} + 6 \cdot 10^{-4}$
    \item $46 \cdot 10^{-3}$
  \end{enumerate}

  \item \small The population of a cell grows to $34,567$. Write this number in standard form
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3.4567 \cdot 10^4$
    \item $3.4567 \cdot 10^5$
    \item $3.4567 \cdot 10^3$
    \item $345.67$
    \item $34567$
  \end{enumerate}

  \item \small A doctor prescribes $324.9$ milligrams of a certain medication to a patient. Round to the nearest $10$ milligrams
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $320$
    \item $330$
    \item $324$
    \item $324.9$
    \item $325$
  \end{enumerate}

  \item \small A lab contains $0.000543$ liters of a chemical solution. Write this number in expanded form
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5 \cdot 10^{-4} + 4 \cdot 10^{-5} + 3 \cdot 10^{-6}$
    \item $5 \cdot 10^{-1} + 4 \cdot 10^{-2} + 3 \cdot 10^{-3}$
    \item $5 \cdot 10^{-3} + 4 \cdot 10^{-4} + 3 \cdot 10^{-5}$
    \item $5 \cdot 10^{-4} + 4 \cdot 10^{-5} + 3 \cdot 10^{-6}$
    \item $543 \cdot 10^{-6}$
  \end{enumerate}

  \item \small The length of a bacterium is $0.0025$ millimeters. Write this in standard form
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2.5 \cdot 10^{-3}$
    \item $2.5 \cdot 10^{-2}$
    \item $2.5 \cdot 10^{-4}$
    \item $2.5 \cdot 10^{-1}$
    \item $2.5 \cdot 10^{1}$
  \end{enumerate}

  \item \small A certain medical treatment costs $$2,456.78$. Round the cost to the nearest $100$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2,500$
    \item $2,400$
    \item $2,456.80$
    \item $2,456.78$
    \item $2,457$
  \end{enumerate}

\end{enumerate}
\end{multicols}
