\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A doctor prescribes $15$ pills to a patient, with $3$ pills to be taken daily. How many days will the pills last?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3 days
    \item 4 days
    \item 5 days
    \item 6 days
    \item 7 days
  \end{enumerate}

  \item \small A biologist is studying cell growth. If a cell divides into $2$ daughter cells every $5$ hours, how many cells will there be after $15$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4 cells
    \item 6 cells
    \item 8 cells
    \item 12 cells
    \item 16 cells
  \end{enumerate}

  \item \small What is the remainder when $27$ is divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 1
    \item 2
    \item 3
    \item 6
  \end{enumerate}

  \item \small A medical trial has $120$ participants. If they are divided into groups of $8$, how many groups can be formed?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 12 groups
    \item 15 groups
    \item 16 groups
    \item 20 groups
    \item 24 groups
  \end{enumerate}

  \item \small A pharmaceutical company produces $210$ units of a medication daily. If $5$ units are packaged in each bottle, how many bottles are filled daily?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 40 bottles
    \item 41 bottles
    \item 42 bottles
    \item 45 bottles
    \item 50 bottles
  \end{enumerate}

  \item \small What is the quotient and remainder when $48$ is divided by $7$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 6 R 6
    \item 6 R 5
    \item 7 R 1
    \item 8 R 2
    \item 9 R 3
  \end{enumerate}

  \item \small A researcher is collecting data on the average lifespan of a species. If the data shows that the lifespan is $25$ years, and this value is $5$ more than a multiple of $4$, what is the remainder when the multiple is divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 1
    \item 2
    \item 3
    \item 4
  \end{enumerate}

  \item \small A hospital has $180$ beds. If $15$ beds are allocated to each ward, how many wards can be created?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10 wards
    \item 11 wards
    \item 12 wards
    \item 13 wards
    \item 14 wards
  \end{enumerate}

\end{enumerate}
