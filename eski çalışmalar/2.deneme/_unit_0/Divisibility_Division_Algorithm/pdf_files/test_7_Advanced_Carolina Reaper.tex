% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A bookshelf has $15$ shelves, and each shelf can hold $18$ books. If the bookshop owner wants to pack the books in boxes that hold $3$ books each, how many boxes can be filled with $15 \times 18$ books?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 90
    \item 135
    \item 150
    \item 180
    \item 270
  \end{enumerate}

  \item \small What is the remainder when $43$ is divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 1
    \item 2
    \item 4
    \item 7
  \end{enumerate}

  \item \small If a bakery sells $252$ loaves of bread per day, and they are packed in bags of $7$ loaves each, how many bags are sold per day?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 35
    \item 36
    \item 42
    \item 45
    \item 63
  \end{enumerate}

  \item \small A group of friends want to share some candy equally. If they have $48$ pieces of candy and there are $8$ friends, how many pieces will each friend get, and what will be the remainder?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5 pieces, remainder 8
    \item 6 pieces, remainder 0
    \item 6 pieces, remainder 1
    \item 5 pieces, remainder 4
    \item 6 pieces, remainder 3
  \end{enumerate}

  \item \small What is the remainder when $2^{10}$ is divided by $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 1
    \item 2
    \item 3
    \item 4
  \end{enumerate}

  \item \small A water tank can hold $240$ gallons of water. If $3$ buckets of water are poured into the tank per hour, and each bucket holds $5$ gallons, what fraction of the tank will be filled after $8$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{2}$
    \item $\frac{1}{3}$
    \item $\frac{2}{3}$
    \item $\frac{3}{4}$
    \item $\frac{4}{5}$
  \end{enumerate}

  \item \small A clock strikes $6$ times at $12$ o'clock. If it strikes $x$ times after $y$ hours, find $x$ when $y = 9$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 18
    \item 15
    \item 12
    \item 9
    \item 6
  \end{enumerate}

  \item \small What is the remainder when $17$ is divided by $5$, and what is the quotient?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Remainder: 2, Quotient: 3
    \item Remainder: 3, Quotient: 3
    \item Remainder: 2, Quotient: 4
    \item Remainder: 1, Quotient: 3
    \item Remainder: 2, Quotient: 2
  \end{enumerate}

\end{enumerate}
\end{multicols}
