% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A hiker travels $24 \cdot 5$ miles in a day. What is the remainder when this distance is divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 3
    \item 6
    \item 9
    \item 12
  \end{enumerate}

  \item \small Which number is divisible by $2$ and $5$ but not by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 15
    \item 20
    \item 25
    \item 30
  \end{enumerate}

  \item \small A park ranger tracks the population of deer. If the population increases by $150$ deer per year, what is the remainder when the total population after $7$ years is divided by $10$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 2
    \item 5
    \item 7
    \item 9
  \end{enumerate}

  \item \small What is the result of $48 \div 6$ with remainder?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8$ R 0
    \item 7$ R 4
    \item 6$ R 6
    \item 8$ R 2
    \item 6$ R 0
  \end{enumerate}

  \item \small A fuel tank can hold $240$ gallons. If $120$ gallons are already filled, what fraction of the tank is filled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{4}$
    \item $\frac{1}{2}$
    \item $\frac{2}{3}$
    \item $\frac{3}{4}$
    \item $\frac{1}{1}$
  \end{enumerate}

  \item \small A mountain trail is $54$ km long. What is the remainder when the length of the trail is divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 3
    \item 6
    \item 9
    \item 12
  \end{enumerate}

  \item \small If a bird sanctuary has $135$ birds and $15$ more birds arrive, what is the remainder when the new total is divided by $10$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 2
    \item 5
    \item 7
    \item 9
  \end{enumerate}

  \item \small A group of friends want to share some candy equally. If they have $48$ pieces of candy and there are $8$ friends, what is the remainder when the candy is divided?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 2
    \item 4
    \item 6
    \item 8
  \end{enumerate}

\end{enumerate}
\end{multicols}
