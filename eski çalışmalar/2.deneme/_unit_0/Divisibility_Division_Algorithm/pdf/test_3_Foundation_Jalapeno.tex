\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A swimmer completed $240$ laps in a pool. Perform long division to find the quotient and remainder when $240$ is divided by $16$. Express your answer in the form $q \cdot d + r$, where $q$ is the quotient, $d$ is the divisor, and $r$ is the remainder. \vspace{2cm}

  \item \small A triathlete ran $180$ kilometers, cycled $90$ kilometers, and swam $45$ kilometers in a competition. What is the remainder when the total distance traveled ($180 + 90 + 45$) is divided by $9$? Show all work and express your answer as a numerical value. \vspace{2cm}

\end{enumerate}
