% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A building has $24$ floors. If $3$ elevators can carry people to every $4^{th}$ floor, how many stops will each elevator make?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 6
    \item 8
    \item 12
    \item 16
  \end{enumerate}

  \item \small A recipe for making cookies requires $15$ cups of flour. If you want to make $3$ batches of cookies, how many cups of flour will you need in total, if $3$ cups can be bought in $1$ package?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 10
    \item 12
    \item 15
    \item 45
  \end{enumerate}

  \item \small If a number is divisible by $2$ and $9$, it must also be divisible by...
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3
    \item 4
    \item 5
    \item 6
    \item 18
  \end{enumerate}

  \item \small What is the remainder when $43$ is divided by $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 1
    \item 2
    \item 3
    \item 3
  \end{enumerate}

  \item \small In a bookshelf with $36$ books, if the books are to be packed in boxes that hold $4$ books each, how many boxes are needed?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8
    \item 9
    \item 10
    \item 12
    \item 6
  \end{enumerate}

  \item \small The sum of the digits of a number is $9$. If the number is divisible by $3$, it must also be divisible by...
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 9
    \item 6
    \item 5
    \item 10
    \item 8
  \end{enumerate}

  \item \small A floor is $12$ meters long. If a carpet that is $3$ meters long is laid along its length, what fraction of the floor will be covered?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{2}$
    \item $\frac{1}{3}$
    \item $\frac{1}{4}$
    \item $\frac{3}{4}$
    \item $\frac{1}{6}$
  \end{enumerate}

  \item \small What is the result of $27 \div 3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$ remainder $3$
    \item $9$
    \item $10$
    \item $12$
    \item $6$ remainder $1$
  \end{enumerate}

\end{enumerate}
