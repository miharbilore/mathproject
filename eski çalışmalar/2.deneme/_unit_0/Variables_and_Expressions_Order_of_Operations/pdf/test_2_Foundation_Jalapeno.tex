  \item \small What is the value of $2 \cdot 3 + 12 \div 4$?

    \item 8
    \item 10
    \item 12
    \item 14
    \item 16
  \end{enumerate}

  \item \small Translate the verbal phrase '5 more than 3 times a number' into an algebraic expression.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3x + 5$
    \item $3x - 5$
    \item $5x + 3$
    \item $5x - 3$
    \item $x + 15$
  \end{enumerate}

  \item \small Simplify the expression $2x + 5 + 3x - 2$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5x + 3$
    \item $5x - 3$
    \item $2x + 3$
    \item $2x - 3$
    \item $x + 7$
  \end{enumerate}

  \item \small Evaluate the expression $x^2 + 4x - 5$ when $x = 2$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3
    \item 5
    \item 7
    \item 9
    \item 11
  \end{enumerate}

  \item \small What is the value of $(8 - 3) \cdot (2 + 1)$?

    \item 10
    \item 12
    \item 15
    \item 18
    \item 20
  \end{enumerate}

  \item \small A space probe travels $3x + 2$ meters in $x$ seconds. If $x = 4$, how many meters will it travel?

    \item 10
    \item 12
    \item 14
    \item 16
    \item 18
  \end{enumerate}

  \item \small The temperature of a planet is $-2x + 10$ degrees Celsius. If $x = 3$, what is the temperature?

    \item 2
    \item 4
    \item 6
    \item 4
    \item 6
  \end{enumerate}
