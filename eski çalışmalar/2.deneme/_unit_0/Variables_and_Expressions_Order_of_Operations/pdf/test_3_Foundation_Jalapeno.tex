% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the value of $2 \cdot 3 + 12 \div 4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 12
    \item 9
    \item 7
    \item 8
  \end{enumerate}

  \item \small Evaluate $18 - 3 + 2$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 17
    \item 16
    \item 15
    \item 13
    \item 12
  \end{enumerate}

  \item \small A tennis player wins $3$ sets in a row. If she wins $2$ points for every set she wins, how many points does she have after $x$ sets?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x + 3$
    \item $2x - 3$
    \item $3x + 2$
    \item $3x - 2$
    \item $6x + 1$
  \end{enumerate}

  \item \small A soccer team scores $2$ goals per game. If they play $x$ games, how many goals will they have scored in total?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x + 1$
    \item $2x - 1$
    \item $x + 2$
    \item $x - 2$
    \item $2x$
  \end{enumerate}

  \item \small Simplify the expression $3(2x - 1) + 2$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6x - 1$
    \item $6x + 1$
    \item $6x - 3$
    \item $6x + 3$
    \item $5x + 2$
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $2x + 5 = 11$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 4
    \item 5
    \item 6
  \end{enumerate}

  \item \small A basketball player scores $15$ points in the first quarter and $8$ points in the second quarter. If $x$ represents the number of points she scores in the third quarter, what is the expression for her total points scored after three quarters?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x + 23$
    \item $x - 23$
    \item $2x + 15$
    \item $2x - 15$
    \item $x + 8$
  \end{enumerate}

  \item \small What is the simplified form of the expression $2x + 3 + 4x - 2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6x + 1$
    \item $6x - 1$
    \item $5x + 2$
    \item $5x - 2$
    \item $6x + 5$
  \end{enumerate}

\end{enumerate}
\end{multicols}
