% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A shipping company charges $2 per package plus $0.50 per mile. If $x$ represents the number of miles, what is the algebraic expression for the total cost?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0.50x + 2$
    \item $2x + 0.50$
    \item $2x - 0.50$
    \item $0.50 - 2x$
    \item $2 - 0.50x$
  \end{enumerate}

  \item \small Evaluate the expression $3 \cdot 2 + 11 - 5$ using the correct order of operations.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10$
    \item $15$
    \item $20$
    \item $6$
    \item $4$
  \end{enumerate}

  \item \small A traffic flow model uses the equation $t = \frac{d}{r}$, where $t$ is time, $d$ is distance, and $r$ is rate. If $d = 250$ miles and $r = 50$ miles per hour, what is the value of $t$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ hours
    \item $3$ hours
    \item $2$ hours
    \item $8$ hours
    \item $10$ hours
  \end{enumerate}

  \item \small Simplify the algebraic expression $2x + 5x - 3x + 2$ by combining like terms.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4x + 2$
    \item $6x - 2$
    \item $4x - 2$
    \item $6x + 2$
    \item $8x - 1$
  \end{enumerate}

  \item \small A delivery company uses the equation $c = 2w + 5$ to calculate the total cost, where $c$ is the total cost and $w$ is the weight of the package in pounds. If the weight of the package is $10$ pounds, what is the total cost?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $25$
    \item $20$
    \item $15$
    \item $10$
    \item $5$
  \end{enumerate}

  \item \small Evaluate the expression $(3 + 2) \cdot (8 - 3)$ using the correct order of operations.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $30$
    \item $25$
    \item $15$
    \item $5$
  \end{enumerate}

  \item \small The equation $d = rt$ represents the distance traveled by a vehicle, where $d$ is distance, $r$ is rate, and $t$ is time. If $r = 40$ miles per hour and $t = 2.5$ hours, what is the distance traveled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $100$ miles
    \item $80$ miles
    \item $120$ miles
    \item $60$ miles
    \item $50$ miles
  \end{enumerate}

  \item \small Simplify the algebraic expression $x^2 + 4x - 3x - 2$ by combining like terms.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 + x - 2$
    \item $x^2 - 3x + 2$
    \item $x^2 + 2x - 1$
    \item $x^2 - x - 2$
    \item $x^2 + 3x - 2$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A logistics company has a fleet of trucks with different capacities. The total capacity of the trucks can be represented by the equation $C = 2x^2 + 5x - 3$, where $C$ is the total capacity and $x$ is the number of trucks. Find the total capacity when there are $4$ trucks. Show your work and explain your answer. \vspace{2cm}

  \item \small A traffic engineer is designing a new highway system. The time it takes for a car to travel from point A to point B can be represented by the equation $t = \frac{d}{r}$, where $t$ is time, $d$ is distance, and $r$ is rate. If the distance between point A and point B is $120$ miles and the rate is $40$ miles per hour, find the time it takes for a car to travel from point A to point B. Additionally, if the rate increases by $10$ miles per hour, how much time is saved? \vspace{2cm}

\end{enumerate}
