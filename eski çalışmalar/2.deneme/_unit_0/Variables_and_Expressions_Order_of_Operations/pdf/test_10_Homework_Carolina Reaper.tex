% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A rhythm pattern has $3 \cdot 2 + 11$ beats. How many beats are in the pattern?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 15
    \item 17
    \item 20
    \item 25
  \end{enumerate}

  \item \small The area of a canvas is $15 - 3 + 2$ square feet. What is the area?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 12
    \item 14
    \item 16
    \item 20
  \end{enumerate}

  \item \small A music frequency is given by $2x + 5$ Hz, where $x = 2$. Find the frequency.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4 Hz
    \item 5 Hz
    \item 9 Hz
    \item 12 Hz
    \item 15 Hz
  \end{enumerate}

  \item \small Simplify the expression $12 - 2 \cdot 3 + 1$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 7
    \item 9
    \item 11
    \item 13
    \item 15
  \end{enumerate}

  \item \small A painter needs $5x - 2$ brushes, where $x = 3$. How many brushes are needed?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 11
    \item 13
    \item 15
    \item 17
    \item 19
  \end{enumerate}

  \item \small The length of a rhythm bar is $18 \div 3 + 2$ inches. Find the length.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 6
    \item 7
    \item 8
    \item 9
    \item 10
  \end{enumerate}

  \item \small Simplify $9 - 3 \cdot 2 + 1$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 4
    \item 6
    \item 8
    \item 10
  \end{enumerate}

  \item \small A musician plays $4 \cdot x + 2$ notes, where $x = 4$. How many notes are played?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 16
    \item 18
    \item 20
    \item 22
    \item 24
  \end{enumerate}

\end{enumerate}
\end{multicols}
