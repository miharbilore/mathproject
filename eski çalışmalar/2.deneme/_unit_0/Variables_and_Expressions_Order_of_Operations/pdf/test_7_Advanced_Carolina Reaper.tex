% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the value of $2 \times 3 + 11$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 17
    \item 20
    \item 21
    \item 25
  \end{enumerate}

  \item \small Simplify the expression $5(2x - 3) + 2(x + 1)$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10x - 13$
    \item $7x - 11$
    \item $7x + 1$
    \item $10x + 2$
    \item $7x - 13$
  \end{enumerate}

  \item \small Evaluate the expression $(18 - 3) \cdot (7 - 2)$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 40
    \item 50
    \item 60
    \item 70
    \item 75
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $2x + 5 = 11$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 4
    \item 5
    \item 6
  \end{enumerate}

  \item \small A bookstore is having a sale where all books are 15% off. If a book originally costs $25, how much will it cost during the sale?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20.25
    \item $21.25
    \item $22.25
    \item $23.25
    \item $24.25
  \end{enumerate}

  \item \small What is the simplified form of $3(2x + 1) + 2(4x - 2)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10x + 1$
    \item $10x - 1$
    \item $10x + 3$
    \item $10x - 3$
    \item $14x - 1$
  \end{enumerate}

  \item \small A bus travels from City A to City B at an average speed of 40 miles per hour. If the distance between the cities is 160 miles, how many hours will the trip take?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3 hours
    \item 4 hours
    \item 5 hours
    \item 6 hours
    \item 7 hours
  \end{enumerate}

  \item \small Simplify the expression $(9 - 2) \cdot (11 + 2)$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 70
    \item 77
    \item 91
    \item 112
    \item 121
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small Tom has been saving money for a new bike and has $120 in his savings account. He wants to buy a bike that costs $180. Tom's parents have agreed to give him an allowance of $5 per week for doing chores. How many weeks will it take Tom to have enough money to buy the bike? Show your work and explain your reasoning. \vspace{2cm}

  \item \small A water tank can hold 1200 gallons of water. Due to a leak, the tank is losing water at a rate of 5 gallons per hour. If 300 gallons of water are already in the tank, how many hours will it take for the tank to be empty? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
