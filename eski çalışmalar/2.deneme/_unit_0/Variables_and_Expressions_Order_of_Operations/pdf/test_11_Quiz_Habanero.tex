\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Evaluate the expression $3 \cdot 2 + 11 - 4$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 14
    \item 15
    \item 16
    \item 17
    \item 18
  \end{enumerate}

  \item \small What is the value of $x$ in the expression $x + 5 = 11$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 7
    \item 8
    \item 9
  \end{enumerate}

  \item \small The temperature in a city is given by $T = 2h - 5$, where $h$ is the number of hours since midnight. Find $T$ when $h = 4$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3
    \item -1
    \item 3
    \item 5
    \item -3
  \end{enumerate}

  \item \small The rainfall in a region is given by $R = 0.5x + 2$, where $x$ is the number of days. Find $R$ when $x = 6$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 6
    \item 7
    \item 8
  \end{enumerate}

  \item \small Simplify the expression $2(x + 3) + x - 1$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x + 5$
    \item $3x + 6$
    \item $3x + 5$
    \item $2x + 6$
    \item $x + 5$
  \end{enumerate}

  \item \small The cost of removing pollutants from a lake is given by $C = 200 + 50x$, where $x$ is the number of days. Find $C$ when $x = 3$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 350
    \item 400
    \item 500
    \item 550
    \item 600
  \end{enumerate}

  \item \small The population of a species is given by $P = 100 + 20t - 5t^2$, where $t$ is the number of years. Find $P$ when $t = 2$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 100
    \item 120
    \item 140
    \item 160
    \item 180
  \end{enumerate}

  \item \small Evaluate the expression $(2 + 3) \cdot (4 - 1)$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 12
    \item 14
    \item 16
    \item 18
  \end{enumerate}

\end{enumerate}
\end{multicols}
