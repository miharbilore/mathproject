\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the value of $3 \cdot 2 + 11$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 17
    \item 19
    \item 20
    \item 21
  \end{enumerate}

  \item \small Simplify $2x + 5 + 3x - 2$. 
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5x + 3$
    \item $5x - 3$
    \item $5x + 7$
    \item $x + 7$
    \item $x - 7$
  \end{enumerate}

  \item \small Evaluate $12 - 3 + 2 \cdot 4$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 13
    \item 15
    \item 17
    \item 19
    \item 21
  \end{enumerate}

  \item \small A gamer has $120$ points. After a game, $2x + 5$ points are added. If $x = 7$, how many points does the gamer have now?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 139
    \item 149
    \item 159
    \item 169
    \item 179
  \end{enumerate}

  \item \small A coding challenge has a scoreboard with $x$ players. If $3x - 2$ players are online and $x = 9$, how many players are online?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 23
    \item 25
    \item 27
    \item 29
    \item 31
  \end{enumerate}

  \item \small Simplify $(7 - 3) \cdot 2$. 
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 6
    \item 7
    \item 8
    \item 10
    \item 12
  \end{enumerate}

  \item \small What is the value of $(4 + 2) \cdot 3 - 1$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 17
    \item 20
    \item 23
    \item 25
    \item 27
  \end{enumerate}

  \item \small A data plan costs $$15$ per month plus $$0.05$ per byte. If $x$ bytes are used, what is the total cost of the plan?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0.05x + 15$
    \item $0.05x - 15$
    \item $15x + 0.05$
    \item $15x - 0.05$
    \item $15 + 0.05x$
  \end{enumerate}

\end{enumerate}
