% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the GCF of $12$ and $18$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A construction worker needs to mix concrete in batches of $8$ or $12$ cubic feet. What is the LCM of these batch sizes?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$
    \item $12$
    \item $24$
    \item $32$
    \item $48$
  \end{enumerate}

  \item \small A recipe for making cookies calls for $24$ ounces of flour. If you want to make half a batch, what is the GCF of $24$ and $2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $6$
    \item $8$
  \end{enumerate}

  \item \small What is the LCM of $9$ and $15$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $30$
    \item $45$
    \item $60$
    \item $90$
  \end{enumerate}

  \item \small A builder is constructing a house with a roof that needs to be $18$ feet long and $12$ feet wide. What is the GCF of these dimensions?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A chef is making a sauce that requires $20$ ounces of ketchup and $30$ ounces of mayonnaise. What is the LCM of these quantities?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $30$
    \item $60$
    \item $90$
    \item $120$
  \end{enumerate}

  \item \small What is the GCF of $48$ and $60$ using factors?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4$
    \item $6$
    \item $8$
    \item $12$
    \item $24$
  \end{enumerate}

  \item \small A carpenter is building a bookshelf with $16$ shelves, each $10$ inches wide. What is the LCM of $16$ and $10$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $16$
    \item $32$
    \item $40$
    \item $80$
    \item $160$
  \end{enumerate}

\end{enumerate}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A contractor is paving a driveway that is $15$ yards long and $8$ yards wide. If the contractor can only lay paving stones in quantities that are multiples of $3$ or $5$, what is the smallest number of stones the contractor needs to buy to cover the entire driveway? Show your work and explain your reasoning. \vspace{2cm}

  \item \small A recipe for making bread calls for $3 \frac{1}{4}$ cups of flour and $2 \frac{1}{2}$ cups of water. If you want to make $2$ batches of bread, what is the total amount of flour and water needed? Find the LCM of the denominators of these fractions and use it to add the fractions. Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
