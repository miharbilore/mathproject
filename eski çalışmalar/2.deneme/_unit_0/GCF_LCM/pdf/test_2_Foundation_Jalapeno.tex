% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the GCF of $12$ and $18$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $9$
  \end{enumerate}

  \item \small A space probe travels $30$ miles in $5$ minutes. How many miles will it travel in $15$ minutes if it maintains the same GCF of speed?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $60$ miles
    \item $90$ miles
    \item $120$ miles
    \item $150$ miles
    \item $180$ miles
  \end{enumerate}

  \item \small What is the LCM of $8$ and $12$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $24$
    \item $32$
    \item $40$
    \item $48$
    \item $56$
  \end{enumerate}

  \item \small A lab experiment requires $15$ ml of a chemical. If the chemical is only available in bottles of $3$ ml or $5$ ml, what is the LCM of the bottle sizes?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$ ml
    \item $30$ ml
    \item $45$ ml
    \item $60$ ml
    \item $75$ ml
  \end{enumerate}

  \item \small What is the GCF of $24$ and $30$ using factors?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A space station has two modules with $18$ and $24$ solar panels. What is the GCF of the number of solar panels?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $9$
  \end{enumerate}

  \item \small A physics experiment involves a pendulum with a length of $12$ inches and a swing period of $8$ seconds. What is the LCM of the length and period?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $24$
    \item $32$
    \item $40$
    \item $48$
    \item $56$
  \end{enumerate}

  \item \small What is the LCM of $9$ and $12$ using multiples?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $18$
    \item $27$
    \item $36$
    \item $45$
    \item $54$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A lab requires $12$ liters of a chemical to conduct an experiment. If the chemical is only available in bottles of $4$ liters or $6$ liters, determine the number of each type of bottle needed to obtain exactly $12$ liters using GCF and LCM. \vspace{2cm}

  \item \small A space mission requires a spacecraft to travel to two planets with orbital periods of $15$ days and $20$ days. Determine the LCM of the orbital periods and explain how this value is used to plan the mission. \vspace{2cm}
