% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Find the GCF of $12$ and $18$ using factors.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small Find the LCM of $9$ and $12$ using multiples.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $18$
    \item $24$
    \item $27$
    \item $36$
    \item $48$
  \end{enumerate}

  \item \small A basketball player scores $15$ points in one game and $20$ points in another. Find the GCF of these scores.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $3$
    \item $5$
    \item $10$
    \item $15$
  \end{enumerate}

  \item \small A cyclist rides $8$ miles in one hour and $12$ miles in another hour. Find the LCM of these distances.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $16$
    \item $20$
    \item $24$
    \item $32$
    \item $40$
  \end{enumerate}

  \item \small Find the prime factorization of $48$ to determine its GCF with $18$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2 \cdot 2 \cdot 2 \cdot 2 \cdot 3$
    \item $2 \cdot 2 \cdot 3 \cdot 3$
    \item $2 \cdot 3 \cdot 3 \cdot 3$
    \item $2 \cdot 2 \cdot 2 \cdot 3 \cdot 3$
    \item $2 \cdot 3 \cdot 3 \cdot 3 \cdot 3$
  \end{enumerate}

  \item \small A football team plays $10$ games in one season and $15$ games in another. Find the LCM of these game counts.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $25$
    \item $30$
    \item $50$
    \item $75$
  \end{enumerate}

  \item \small A tennis player wins $9$ sets in one tournament and $12$ sets in another. Find the GCF of these set counts.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $9$
  \end{enumerate}

  \item \small Find the LCM of $8$ and $15$ using prime factorization.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $30$
    \item $40$
    \item $60$
    \item $120$
    \item $240$
  \end{enumerate}

\end{enumerate}
\end{multicols}
