% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the decimal equivalent of the fraction $\frac{3}{4}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.25
    \item 0.5
    \item 0.75
    \item 0.8
    \item 0.9
  \end{enumerate}

  \item \small A space station is traveling at $\frac{3}{5}$ of the speed of light. What percent is this?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20%
    \item 30%
    \item 40%
    \item 50%
    \item 60%
  \end{enumerate}

  \item \small Which of the following rational numbers is the largest: $\frac{1}{2}$, $0.4$, or $\frac{3}{5}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{2}$
    \item 0.4
    \item $\frac{3}{5}$
    \item They are equal
    \item None of the above
  \end{enumerate}

  \item \small A sample of Moon rock weighs $2.5$ kg. What fraction of a kilogram is $0.8$ kg?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{4}$
    \item $\frac{1}{3}$
    \item $\frac{2}{5}$
    \item $\frac{3}{5}$
    \item $\frac{2}{3}$
  \end{enumerate}

  \item \small Order the rational numbers $\frac{2}{3}$, $0.6$, and $\frac{3}{4}$ from least to greatest.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{2}{3}$, $0.6$, $\frac{3}{4}$
    \item $0.6$, $\frac{2}{3}$, $\frac{3}{4}$
    \item $\frac{3}{4}$, $\frac{2}{3}$, $0.6$
    \item $0.6$, $\frac{3}{4}$, $\frac{2}{3}$
    \item $\frac{2}{3}$, $\frac{3}{4}$, $0.6$
  \end{enumerate}

  \item \small The cost of launching a spacecraft is $\frac{3}{4}$ million dollars. What is this amount in decimal form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.25 million
    \item 0.5 million
    \item 0.75 million
    \item 0.8 million
    \item 0.9 million
  \end{enumerate}

  \item \small A physics experiment measures a distance of $2.4$ meters. What fraction of a meter is $0.8$ meters?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{3}$
    \item $\frac{1}{4}$
    \item $\frac{1}{2}$
    \item $\frac{2}{3}$
    \item $\frac{3}{4}$
  \end{enumerate}

  \item \small A spaceship travels $\frac{2}{3}$ of the distance to Mars. What percent of the distance is this?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20%
    \item 30%
    \item 40%
    \item 50%
    \item 66.67%
  \end{enumerate}

\end{enumerate}
\end{multicols}

\end{enumerate}
