% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A music producer wants to adjust the tempo of a song by $25\%$. If the original tempo is $120$ beats per minute, what is the new tempo in beats per minute?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 140
    \item 150
    \item 160
    \item 130
    \item 110
  \end{enumerate}

  \item \small A painter is creating a mural on a canvas that is $15 \frac{3}{4}$ feet long. What is the length of the canvas in decimal form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15.5
    \item 15.75
    \item 16.25
    \item 15.25
    \item 16.5
  \end{enumerate}

  \item \small A rhythm pattern consists of $6$ eighth-note beats. If the time signature is $\frac{3}{4}$, how many quarter-note beats are in the pattern?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 6
    \item 8
    \item 9
    \item 12
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $2.5x = 10 \frac{1}{4}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3
    \item 4
    \item 4.1
    \item 4.05
    \item 3.5
  \end{enumerate}

  \item \small Which of the following rational numbers is equivalent to $0.6 \overline{3}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{2}{3}$
    \item $\frac{3}{5}$
    \item $\frac{5}{8}$
    \item $\frac{7}{11}$
    \item $\frac{8}{13}$
  \end{enumerate}

  \item \small A musician is playing a guitar solo that consists of a series of $8$-note phrases. If the first phrase starts on beat $2 \frac{1}{4}$, which beat will the final note of the phrase fall on?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 10.25
    \item 9.75
    \item 10.5
    \item 9.5
  \end{enumerate}

  \item \small A painter is creating a series of $6$ identical paintings, each with a border that is $1 \frac{3}{8}$ inches wide. What is the total width of the borders in decimal form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8
    \item 8.25
    \item 9.5
    \item 8.5
    \item 10
  \end{enumerate}

  \item \small A music producer wants to create a song that is $3 \frac{1}{2}$ minutes long. If the song is divided into $6$ equal sections, how long is each section in decimal form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.55
    \item 0.58
    \item 0.5
    \item 0.6
    \item 0.65
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}

  \item \small A painting is $12 \frac{3}{4}$ feet long and $8 \frac{1}{2}$ feet wide. If the frame around the painting is $1 \frac{1}{4}$ inches wide, what is the total area of the painting and frame in square feet? Show your work and be sure to convert all measurements to decimal form before calculating the area. \vspace{2cm}

  \item \small A musician is playing a $16$-bar guitar solo in $\frac{4}{4}$ time. If the first $8$ bars are in the key of $A$ and the last $8$ bars are in the key of $E$, what is the total time spent in each key in decimal form? Assume the tempo is $120$ beats per minute and show your work. \vspace{2cm}

\end{enumerate}
