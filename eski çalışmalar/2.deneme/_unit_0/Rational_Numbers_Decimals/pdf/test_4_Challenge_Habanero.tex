% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A recipe for cake requires $2 \frac{3}{4}$ cups of flour. If you want to make half the recipe, how many cups of flour will you need?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1 \frac{3}{8}$
    \item $1 \frac{3}{4}$
    \item $1 \frac{1}{2}$
    \item $1 \frac{1}{4}$
    \item $1 \frac{5}{8}$
  \end{enumerate}

  \item \small A building is $25.5$ meters tall. If a scale model of the building is $0.5$ meters tall, what is the scale of the model?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1:50$
    \item $1:51$
    \item $1:52$
    \item $1:49$
    \item $1:48$
  \end{enumerate}

  \item \small A chef needs to mix $3 \frac{1}{2}$ liters of sauce with $2 \frac{3}{4}$ liters of water. How many liters of mixture will the chef have in total?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6 \frac{1}{4}$
    \item $6 \frac{1}{2}$
    \item $6 \frac{3}{4}$
    \item $7 \frac{1}{4}$
    \item $7 \frac{1}{2}$
  \end{enumerate}

  \item \small A carpenter has a plank that is $5.8$ meters long. If the carpenter cuts $2.1$ meters from the plank, what fraction of the plank remains?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{3}{5}$
    \item $\frac{4}{5}$
    \item $\frac{7}{10}$
    \item $\frac{3}{8}$
    \item $\frac{7}{8}$
  \end{enumerate}

  \item \small A recipe requires $1.25$ cups of oil. If you only have a $\frac{1}{4}$ cup measuring cup, how many times will you need to fill the measuring cup?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 6
    \item 8
    \item 10
  \end{enumerate}

  \item \small A water tank can hold $4500$ liters of water. If $2 \frac{1}{4}$ percent of the tank is already filled, how many liters of water are in the tank?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 90
    \item 100
    \item 101.25
    \item 112.5
    \item 101
  \end{enumerate}

  \item \small A architect designed a building with a roof that is $4.2$ meters above the floor. If a $2.5$ meter ladder is placed against the wall, what fraction of the ladder is above the roof?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{8}$
    \item $\frac{1}{7}$
    \item $\frac{3}{14}$
    \item $\frac{1}{4}$
    \item $\frac{3}{8}$
  \end{enumerate}

  \item \small A painter needs to mix $2.8$ liters of paint with $1.2$ liters of solvent. If the painter adds $0.5$ liters of water, what is the total amount of mixture?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4.5$
    \item $4.3$
    \item $4.1$
    \item $4.7$
    \item $5.5$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A bridge is $235.6$ meters long. If a section of the bridge is $3 \frac{1}{2}$ meters above the water level, and the water level rises by $0.8$ meters, what fraction of the bridge is above the water level? Show your work and provide your answer as a fraction and a decimal. \vspace{2cm}

  \item \small A contractor needs to pour $15.2$ cubic meters of concrete into a foundation. If the concrete is $2.5$ percent water, how many liters of water will be in the concrete? Show your work and provide your answer in liters. \vspace{2cm}

\end{enumerate}
