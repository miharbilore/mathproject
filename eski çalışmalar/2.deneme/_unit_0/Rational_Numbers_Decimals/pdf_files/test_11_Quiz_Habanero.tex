% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small The temperature dropped from $25\degree C$ to $15\degree C$. What is the decrease in temperature as a decimal?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.1
    \item 0.5
    \item 0.9
    \item 0.4
    \item 0.01
  \end{enumerate}

  \item \small A water tank can hold $4500$ liters of water. If $3/4$ of the tank is filled, how many liters of water are in the tank?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2000
    \item 3375
    \item 3000
    \item 4000
    \item 2500
  \end{enumerate}

  \item \small The rainfall in a region was $2.5$ cm. What is this as a fraction?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5/2$
    \item $1/2$
    \item $3/2$
    \item $2/5$
    \item $25/10$
  \end{enumerate}

  \item \small Which of the following rational numbers is greatest: $3/4$, $2/3$, $5/6$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3/4$
    \item $2/3$
    \item $5/6$
    \item $1/2$
    \item $1/3$
  \end{enumerate}

  \item \small A river's water level rose by $0.8$ meters. What is this as a percent of $1$ meter?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8%
    \item 80%
    \item 0.8%
    \item 18%
    \item 28%
  \end{enumerate}

  \item \small A weather balloon was found $2.25$ km away from the launch site. What is this distance as a fraction?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $9/4$
    \item $5/2$
    \item $2/5$
    \item $45/20$
    \item $18/8$
  \end{enumerate}

  \item \small A climate system's temperature decreased from $30\degree C$ to $-5\degree C$. What is the change in temperature as a decimal?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item -35.0
    \item 35.0
    \item -0.35
    \item 0.35
    \item -3.5
  \end{enumerate}

  \item \small A park's lake level rose by $1/5$ meters. What is this as a decimal?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.1
    \item 0.2
    \item 0.5
    \item 0.05
    \item 0.01
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A city's average temperature decreased by $2.5\degree C$ over $5$ years. If the initial temperature was $25\degree C$, what was the final temperature in Celsius? Show your work and express your answer as a decimal. \vspace{2cm}

  \item \small The ocean's water level rose by $0.85$ meters due to climate change. What is the percent increase if the initial water level was $5$ meters? Use $3$ significant figures and show your work. \vspace{2cm}

\end{enumerate}
