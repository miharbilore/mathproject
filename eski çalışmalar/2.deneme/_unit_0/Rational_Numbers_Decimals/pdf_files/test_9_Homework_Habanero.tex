% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A patient needs $0.5$ mg of a medicine per day. If the medicine is available in $0.25$ mg tablets, how many tablets should the patient take daily?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1
    \item 2
    \item 3
    \item 4
    \item 5
  \end{enumerate}

  \item \small The growth rate of a cell is $25%$ per hour. If there are $100$ cells initially, how many cells will there be after $3$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 100
    \item 125
    \item 156.25
    \item 195.31
    \item 244.14
  \end{enumerate}

  \item \small What is the decimal equivalent of the fraction $\frac{3}{8}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.25
    \item 0.33
    \item 0.375
    \item 0.4
    \item 0.5
  \end{enumerate}

  \item \small Which of the following rational numbers is greatest: $0.4$, $0.04$, $0.25$, $0.5$, $0.01$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.01
    \item 0.04
    \item 0.25
    \item 0.4
    \item 0.5
  \end{enumerate}

  \item \small A medicine bottle contains $500$ ml of liquid. If $0.2$ ml of the medicine is to be given to a patient, what fraction of the total liquid is given?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{100}$
    \item $\frac{1}{250}$
    \item $\frac{1}{500}$
    \item $\frac{1}{1000}$
    \item $\frac{1}{2000}$
  \end{enumerate}

  \item \small The probability of a patient recovering from a disease is $80%$. If $50$ patients are treated, how many are expected to recover?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 30
    \item 35
    \item 40
    \item 45
    \item 50
  \end{enumerate}

  \item \small What is $2.5 \cdot 0.4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.8
    \item 1
    \item 1.2
    \item 1.5
    \item 2
  \end{enumerate}

  \item \small A bacteria population doubles every $2$ hours. If there are $200$ bacteria initially, how many will there be after $6$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 800
    \item 1600
    \item 2400
    \item 3200
    \item 6400
  \end{enumerate}

\end{enumerate}
