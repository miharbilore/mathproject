% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the decimal equivalent of the fraction $\frac{3}{4}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.25
    \item 0.5
    \item 0.75
    \item 0.8
    \item 0.9
  \end{enumerate}

  \item \small A stock's price increased by $2.50. If this represents a $\frac{1}{4}$ increase, what was the original price?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5
    \item $8
    \item $10
    \item $12
    \item $15
  \end{enumerate}

  \item \small Order the following rational numbers from least to greatest: $-0.5, \frac{1}{3}, 0.25$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-0.5, 0.25, \frac{1}{3}$
    \item $-0.5, \frac{1}{3}, 0.25$
    \item $0.25, -0.5, \frac{1}{3}$
    \item $\frac{1}{3}, 0.25, -0.5$
    \item $0.25, \frac{1}{3}, -0.5$
  \end{enumerate}

  \item \small A savings account earns an annual interest rate of $4\%$. If the principal amount is $\$1000$, what is the interest earned in the first year?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20
    \item $30
    \item $40
    \item $50
    \item $60
  \end{enumerate}

  \item \small What is the fraction equivalent of the decimal $0.6$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{3}{5}$
    \item $\frac{2}{3}$
    \item $\frac{3}{4}$
    \item $\frac{1}{2}$
    \item $\frac{2}{5}$
  \end{enumerate}

  \item \small A company's stock price is $58.25. If it increases by $2.75, what is the new price?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $55
    \item $60
    \item $61
    \item $62
    \item $63
  \end{enumerate}

  \item \small A water tank can hold $2400$ gallons of water. If $\frac{3}{4}$ of the tank is filled, how many gallons of water are in the tank?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1200
    \item 1500
    \item 1600
    \item 1800
    \item 2000
  \end{enumerate}

  \item \small What is $2.5 \cdot 1.8$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3.5
    \item 4.0
    \item 4.5
    \item 4.8
    \item 5.0
  \end{enumerate}

\end{enumerate}
\end{multicols}
