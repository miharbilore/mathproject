% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A music store sells guitars for $$\frac{3}{4}$ of the original price. If the original price is $800, how much will you pay?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$600$
    \item $$640$
    \item $$540$
    \item $$580$
    \item $$620$
  \end{enumerate}

  \item \small What is $$0.\overline{45}$$ as a fraction?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$\frac{1}{20}$
    \item $$\frac{5}{11}$
    \item $$\frac{1}{22}$
    \item $$\frac{41}{90}$
    \item $$\frac{4}{9}$
  \end{enumerate}

  \item \small A bakery is having a sale on bread. A loaf that originally costs $$\frac{3}{5}$$ of a dollar is now on sale for $$0.6$$ dollars. What is the percent decrease in price?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$20\%$
    \item $$10\%$
    \item $$5\%$
    \item $$15\%$
    \item $$25\%$
  \end{enumerate}

  \item \small Order the following rational numbers from least to greatest: $$\frac{5}{8}, 0.6, \frac{9}{16}, 0.7$$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$0.6, \frac{5}{8}, \frac{9}{16}, 0.7$$
    \item $$0.6, \frac{9}{16}, \frac{5}{8}, 0.7$$
    \item $$\frac{9}{16}, 0.6, \frac{5}{8}, 0.7$$
    \item $$0.6, \frac{9}{16}, 0.7, \frac{5}{8}$$
    \item $$\frac{5}{8}, \frac{9}{16}, 0.6, 0.7$$
  \end{enumerate}

  \item \small If $$x = 0.\overline{9}$$, then $$x + \frac{1}{9}$$ equals
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$\frac{10}{9}$$
    \item $$1$$
    \item $$\frac{11}{9}$$
    \item $$2$$
    \item $$\frac{12}{9}$$
  \end{enumerate}

  \item \small A water tank can hold $$\frac{3}{4}$$ of a thousand gallons. If $$\frac{1}{6}$$ of the tank is filled, how many gallons of water are in the tank?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$50$$ gallons
    \item $$40$$ gallons
    \item $$60$$ gallons
    \item $$125$$ gallons
    \item $$75$$ gallons
  \end{enumerate}

  \item \small A recipe calls for $$\frac{3}{8}$$ cup of sugar. If you want to make $$\frac{3}{4}$$ of the recipe, how much sugar will you need?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$\frac{1}{8}$$ cup
    \item $$\frac{9}{32}$$ cup
    \item $$\frac{1}{4}$$ cup
    \item $$\frac{7}{32}$$ cup
    \item $$\frac{5}{16}$$ cup
  \end{enumerate}

  \item \small A car rental company charges $$\frac{1}{4}$$ of the total cost for insurance. If the total cost is $$$480$$, how much will you pay for insurance?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$\$120$$
    \item $$\$150$$
    \item $$\$100$$
    \item $$\$90$$
    \item $$\$110$$
  \end{enumerate}

\end{enumerate}
\end{multicols}
