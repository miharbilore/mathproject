% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A hiker travels $\frac{3}{4}$ of a mile in $\frac{1}{2}$ hour. What is her speed in miles per hour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1.5$
    \item $2$
    \item $2.5$
    \item $3$
    \item $3.5$
  \end{enumerate}

  \item \small A park ranger measures the population of deer in a forest. If the population increases by $25\%$ from $120$ deer to $x$ deer, what is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $120$
    \item $130$
    \item $150$
    \item $160$
    \item $180$
  \end{enumerate}

  \item \small A car's fuel tank can hold $\frac{5}{8}$ of a gallon. If it currently has $\frac{3}{8}$ of a gallon, what fraction of the tank is empty?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{8}$
    \item $\frac{1}{4}$
    \item $\frac{1}{2}$
    \item $\frac{3}{4}$
    \item $\frac{5}{8}$
  \end{enumerate}

  \item \small A river is $\frac{2}{3}$ mile wide. If a bridge spans $\frac{1}{4}$ of the river, what is the length of the bridge in miles?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{6}$
    \item $\frac{1}{4}$
    \item $\frac{1}{3}$
    \item $\frac{1}{2}$
    \item $\frac{2}{3}$
  \end{enumerate}

  \item \small A bicycle travels $\frac{5}{6}$ of a mile on $\frac{1}{3}$ gallon of gas. How many miles will it travel on $1$ gallon of gas?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1.5$
    \item $2.5$
    \item $3.5$
    \item $4.5$
    \item $5.5$
  \end{enumerate}

  \item \small A map is drawn to scale with $1$ inch representing $\frac{1}{4}$ mile. If a distance on the map is $2.5$ inches, what is the actual distance in miles?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0.5$
    \item $0.625$
    \item $0.75$
    \item $1$
    \item $1.25$
  \end{enumerate}

  \item \small A water tank can hold $\frac{3}{4}$ of a gallon. If $\frac{1}{8}$ of a gallon is already filled, what fraction of the tank is still empty?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{8}$
    \item $\frac{1}{4}$
    \item $\frac{1}{2}$
    \item $\frac{3}{4}$
    \item $\frac{5}{8}$
  \end{enumerate}

  \item \small A group of friends want to share a trail mix that is $\frac{3}{5}$ nuts and $\frac{2}{5}$ dried fruit. If they have $20$ ounces of trail mix, how many ounces of nuts do they have?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$
    \item $10$
    \item $12$
    \item $14$
    \item $16$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}

  \item \small A hiker is planning a trip that is $12$ miles long. She wants to complete the trip in $4$ hours. If she has already completed $\frac{3}{4}$ of the trip, what is her current speed in miles per hour? Show your work and explain your reasoning. \vspace{2cm}

  \item \small A park ranger is monitoring the population of a certain species of animal. The population is currently $120$ and is increasing at a rate of $15\%$ per year. If the park ranger wants to maintain a population of no more than $180$, how many years can the population increase at this rate? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
