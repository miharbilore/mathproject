% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small In ancient Babylon, a farmer used $\frac{3}{4}$ of his land for wheat. What decimal represents this fraction?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.25
    \item 0.5
    \item 0.75
    \item 0.9
    \item 1.0
  \end{enumerate}

  \item \small The Great Pyramid of Giza's original height is estimated to be $146.5$ meters. What is this height as a percent of $200$ meters?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 50%
    \item 55%
    \item 66%
    \item 73.25%
    \item 80%
  \end{enumerate}

  \item \small A ancient Greek vase has $\frac{1}{8}$ of its height filled with water. If the total height of the vase is $24$ inches, how many inches of water are in the vase?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 4
    \item 6
    \item 8
  \end{enumerate}

  \item \small What is $0.45$ as a fraction in simplest form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{1}{20}$
    \item $\frac{1}{5}$
    \item $\frac{4}{5}$
    \item $\frac{9}{20}$
    \item $\frac{2}{5}$
  \end{enumerate}

  \item \small In a timeline of ancient civilizations, the proportion of time spent by the Indus Valley Civilization is $\frac{2}{5}$. What percent is this?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20%
    \item 30%
    \item 40%
    \item 50%
    \item 60%
  \end{enumerate}

  \item \small The ancient Olympic Games' $400$-meter dash was said to be $\frac{1}{4}$ the length of the $1600$-meter relay. What decimal represents this fraction?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.25
    \item 0.4
    \item 0.5
    \item 0.6
    \item 0.75
  \end{enumerate}

  \item \small A museum has an artifact that is $0.8$ meters long. What fraction of a meter is this in simplest form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{2}{5}$
    \item $\frac{4}{5}$
    \item $\frac{3}{4}$
    \item $\frac{1}{2}$
    \item $\frac{3}{10}$
  \end{enumerate}

  \item \small In ancient Rome, a merchant sold $\frac{3}{10}$ of his goods. What percent is this?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20%
    \item 25%
    \item 30%
    \item 33.33%
    \item 40%
  \end{enumerate}

\end{enumerate}
\end{multicols}
