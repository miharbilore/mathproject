% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the classification of the number $0$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural Number
    \item Whole Number
    \item Integer
    \item All of the above
    \item None of the above
  \end{enumerate}

  \item \small Which of the following numbers is an example of an odd integer?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-2$
    \item $0$
    \item $-1$
    \item $2$
    \item $1$
  \end{enumerate}

  \item \small Identify the next number in the pattern: $2, 5, 8, 11, 14$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $17$
    \item $18$
    \item $20$
    \item $21$
  \end{enumerate}

  \item \small A book costs $ $15$. If you have a $10$% discount coupon, how much will you pay for the book?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $13.50$
    \item $14.50$
    \item $15.50$
    \item $16.50$
    \item $17.50$
  \end{enumerate}

  \item \small Solve the equation: $x + 2 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 5$
    \item $x = 6$
    \item $x = 7$
    \item $x = 8$
    \item $x = 9$
  \end{enumerate}

  \item \small A bakery sells $480$ loaves of bread per day. If they pack $12$ loaves in a box, how many boxes do they sell per day?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $40$ boxes
    \item $48$ boxes
    \item $60$ boxes
    \item $80$ boxes
    \item $90$ boxes
  \end{enumerate}

  \item \small Solve the inequality: $3x - 2 > 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 3$
    \item $x > 2$
    \item $x > 1$
    \item $x < 3$
    \item $x < 2$
  \end{enumerate}

  \item \small Which of the following is equivalent to $-3 cdot -4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-12$
    \item $-7$
    \item $7$
    \item $12$
    \item $14$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A group of friends want to go on a road trip that is $270$ miles long. If they drive at an average speed of $45$ miles per hour, how many hours will it take them to complete the trip? Additionally, if they leave at $8$am, what time will they arrive at their destination? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
