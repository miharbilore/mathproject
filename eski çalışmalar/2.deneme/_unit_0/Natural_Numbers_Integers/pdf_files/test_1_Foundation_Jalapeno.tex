% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Classify the number $5$ as:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural and whole
    \item Integer only
    \item Even
    \item Negative
    \item None of the above
  \end{enumerate}

  \item \small Which of the following is an example of an odd number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$
    \item $4$
    \item $7$
    \item $10$
    \item $12$
  \end{enumerate}

  \item \small A stock price increased by $3$ dollars. If the original price was $\$$25$, what is the new price?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $22$
    \item $28$
    \item $30$
    \item $32$
  \end{enumerate}

  \item \small Identify the next number in the pattern: $2, 5, 8, 11, 14, \ldots$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $17$
    \item $18$
    \item $20$
    \item $22$
  \end{enumerate}

  \item \small What is the result of $(-3) + 5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-8$
    \item $-2$
    \item $2$
    \item $5$
    \item $8$
  \end{enumerate}

  \item \small A tax return resulted in a refund of $-\$200$. What is the meaning of the negative sign?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Owed money
    \item Refund
    \item No balance
    \item Credit
    \item Debit
  \end{enumerate}

  \item \small Which of the following numbers is a whole number but not a natural number?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $-1$
    \item $2$
    \item $-2$
  \end{enumerate}

  \item \small A savings account has a balance of $\$$500$. After a withdrawal of $\$$200$, what is the new balance?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $300$
    \item $400$
    \item $500$
    \item $600$
    \item $700$
  \end{enumerate}

\end{enumerate}
\end{multicols}
