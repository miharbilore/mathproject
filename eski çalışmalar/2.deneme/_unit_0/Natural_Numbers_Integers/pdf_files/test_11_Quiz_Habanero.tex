% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small The temperature in a desert ecosystem rose by $5$ degrees from $-2$ to what value?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $3$
    \item $5$
    \item $7$
    \item $9$
  \end{enumerate}

  \item \small What is the next number in the pattern: $2, 5, 8, 11, 14$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $17$
    \item $18$
    \item $20$
    \item $22$
  \end{enumerate}

  \item \small A rain gauge in a forest ecosystem measured $-3$ inches of rainfall due to evaporation. If $2$ inches of rain fell, what is the new reading?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-5$
    \item $-1$
    \item $1$
    \item $3$
    \item $5$
  \end{enumerate}

  \item \small Classify the number $-10$ as a natural number, whole number, or integer.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural number
    \item Whole number
    \item Integer
    \item Natural and whole number
    \item Whole and integer
  \end{enumerate}

  \item \small The atmospheric pressure in a mountain ecosystem is $800$ millibars. If it decreases by $15$ millibars every hour, what is the pressure after $4$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $755$
    \item $765$
    \item $785$
    \item $795$
    \item $805$
  \end{enumerate}

  \item \small Determine if $-6$ is an even or odd number.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Even
    \item Odd
    \item Neither
    \item Both
    \item  Cannot be determined
  \end{enumerate}

  \item \small A water tank in a rural ecosystem can hold $1000$ gallons of water. If $-200$ gallons are already used, and $50$ gallons are added, how many gallons are left?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $850$
    \item $900$
    \item $950$
    \item $1000$
    \item $1050$
  \end{enumerate}

  \item \small The number of species in a particular ecosystem is $25$. If $5$ more species are discovered, what is the new total?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $25$
    \item $30$
    \item $35$
    \item $40$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A researcher is studying the effect of temperature changes on plant growth. The temperature changed from $-4$ to $2$ degrees over a period of $3$ days. What is the average change in temperature per day? Show your work and explain your answer. \vspace{2cm}

  \item \small A park ranger is monitoring the water level in a lake ecosystem. If the water level rises by $2$ feet every day for $5$ days, and then decreases by $1$ foot on the sixth day, what is the net change in the water level? Use integers to represent the change in the water level and explain your answer. \vspace{2cm}

\end{enumerate}
