\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small The number of planets in our solar system is $8$. What type of number is $8$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]

    \item Natural and even
    \item Integer and odd
    \item Whole and odd
    \item None of the above
  \end{enumerate}

  \item \small The temperature in a lab experiment is $-5$ degrees Celsius. What type of number is $-5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural
    \item Whole
    \item Integer
    \item Even
    \item Odd
  \end{enumerate}

  \item \small The pattern of the number of astronauts on a space mission is $2, 4, 6, 8, ...$. What comes next in the pattern?

    \item $10$
    \item $12$
    \item $14$
    \item $16$
  \end{enumerate}

  \item \small The number of electrons in an atom is $12$. What type of number is $12$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural and odd
    \item Natural and even
    \item Integer and odd
    \item Whole and odd
    \item None of the above
  \end{enumerate}

  \item \small The result of $3 + (-2)$ is $a$. What is the value of $a$?

    \item $3$
    \item $5$
    \item $-1$
    \item $-3$
  \end{enumerate}

    \item Integer and odd
    \item Whole and odd
    \item None of the above
  \end{enumerate}

  \item \small The pattern of the number of satellites in orbit is $1, 3, 5, 7, ...$. What comes next in the pattern?
