% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A symphony has $3 \cdot 4$ movements. How many movements are there in total?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 12
    \item 14
    \item 16
    \item 20
  \end{enumerate}

  \item \small A rhythm pattern repeats every $5$ beats. If it starts on beat $0$, on which beat will it repeat for the $4^{th}$ time?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 16
    \item 19
    \item 20
    \item 21
  \end{enumerate}

  \item \small The frequency of a sound wave is $-2 \cdot 10^{2}$ Hz. Is this frequency odd or even?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Odd
    \item Even
    \item Neither
    \item Both
    \item Cannot Determine
  \end{enumerate}

  \item \small A painter uses $12$ colors to create a canvas. If she adds $-4$ more colors, how many colors does she have now?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8
    \item 10
    \item 12
    \item 16
    \item 20
  \end{enumerate}

  \item \small A musician plays a $3$-note pattern, then a $5$-note pattern. If this sequence repeats, what is the next note in the pattern after $3 + 5 = 8$ notes?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1^{st}$ note of the $3$-note pattern
    \item 2^{nd}$ note of the $3$-note pattern
    \item 3^{rd}$ note of the $3$-note pattern
    \item 1^{st}$ note of the $5$-note pattern
    \item 2^{nd}$ note of the $5$-note pattern
  \end{enumerate}

  \item \small A canvas size is given as $6 \times -2$. What is the area of the canvas?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-12$
    \item $12$
    \item $-6$
    \item $6$
    \item $0$
  \end{enumerate}

  \item \small The integer $-3$ represents a low pitch in music. If this pitch is increased by $2$, what is the new pitch?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-1$
    \item $-2$
    \item $-3$
    \item $-4$
    \item $-5$
  \end{enumerate}

  \item \small A dance has a rhythm pattern of $2, 4, 6, 8, ...$. What comes next in the pattern?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 9
    \item 10
    \item 12
    \item 14
    \item 16
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A piano keyboard has $88$ keys. If a musician plays a $3$-note pattern on the first $3$ keys, and then repeats this pattern on the next $3$ keys, and so on, how many times will the musician have played the pattern after playing $264$ notes? Show your work and explain your reasoning. \vspace{2cm}

  \item \small A symphony has four movements. The first movement has $120$ notes, the second movement has $-60$ notes, the third movement has $2 \cdot 60$ notes, and the fourth movement has $-2 \cdot 30$ notes. What is the total number of notes in the symphony? If the total is negative, what does this represent in the context of music? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
