% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Classify the number $5$ as:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural and whole
    \item Natural but not whole
    \item Whole but not natural
    \item Integer but not natural
    \item None of the above
  \end{enumerate}

  \item \small Identify the next even number in the pattern: $2, 4, 6, 8, 10$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$
    \item $14$
    \item $16$
    \item $18$
    \item $20$
  \end{enumerate}

  \item \small A basketball player scored $-3$ points in a game due to penalties. If they scored $5$ points from shots, what is their total score?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$
    \item $5$
    \item $-3$
    \item $8$
    \item $-8$
  \end{enumerate}

  \item \small A gymnast performs $15$ routines. If $9$ routines are odd-numbered, how many are even-numbered?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$
    \item $6$
    \item $7$
    \item $8$
    \item $9$
  \end{enumerate}

  \item \small The temperature at a sports stadium is $-2$ degrees Celsius. If it rises by $5$ degrees, what is the new temperature?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3$
    \item $5$
    \item $7$
    \item $-2$
    \item $-7$
  \end{enumerate}

  \item \small A cyclist travels $12$ km in $2$ hours. How many kilometers do they travel per hour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4$ km/h
    \item $5$ km/h
    \item $6$ km/h
    \item $8$ km/h
    \item $10$ km/h
  \end{enumerate}

  \item \small A sports team wins $8$ games and loses $-2$ games. What is their total number of games played?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6$
    \item $8$
    \item $10$
    \item $12$
    \item $14$
  \end{enumerate}

  \item \small Identify the pattern in the sequence: $1, 3, 5, 7, 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Adding $1$
    \item Adding $2$
    \item Adding $3$
    \item Multiplying by $2$
    \item Adding $2$ then subtracting $1$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A marathon runner travels $25$ km in $5$ hours. If they travel at a constant speed, how many kilometers will they travel in $8$ hours? Show your calculations and explain your reasoning. \vspace{2cm}

  \item \small A swimmer finishes a race in $2$ minutes and $15$ seconds. If their opponent finished the same race in $2$ minutes and $30$ seconds, how many seconds faster did the swimmer finish? Convert all times to seconds and show your work. \vspace{2cm}

\end{enumerate}
