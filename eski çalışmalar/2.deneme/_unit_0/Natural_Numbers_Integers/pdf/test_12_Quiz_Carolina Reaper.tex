% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A ship is traveling at a speed of $25 \cdot 10^3$ meters per hour. How many meters will it travel in $-2$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-50 \cdot 10^3$ meters
    \item $50 \cdot 10^3$ meters
    \item $-25 \cdot 10^3$ meters
    \item $25 \cdot 10^3$ meters
    \item 0 meters
  \end{enumerate}

  \item \small What is the next number in the pattern: $2, 5, 8, 11, 14$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 17
    \item 18
    \item 20
    \item 21
  \end{enumerate}

  \item \small Classify the number $-17$ as:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural Number
    \item Whole Number
    \item Integer
    \item Rational Number
    \item Irrational Number
  \end{enumerate}

  \item \small A truck driver is $3$ hours behind schedule. If he drives for $-2$ hours, how many hours behind schedule will he be?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1 hour
    \item 3 hours
    \item 5 hours
    \item 6 hours
    \item 7 hours
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $x + 2 = 7$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 7
    \item 8
    \item 9
  \end{enumerate}

  \item \small A traffic light has a cycle of $12$ seconds. If it has been $-4$ seconds since the light turned green, how many seconds until it turns green again?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8 seconds
    \item 10 seconds
    \item 12 seconds
    \item 16 seconds
    \item 20 seconds
  \end{enumerate}

  \item \small A delivery truck is traveling at a speed of $40$ kilometers per hour. If it travels for $-1.5$ hours, how many kilometers will it have traveled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-60$ kilometers
    \item $-50$ kilometers
    \item $-40$ kilometers
    \item $-30$ kilometers
    \item $-20$ kilometers
  \end{enumerate}

  \item \small What is the result of $(-3) \cdot (-6)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-18$
    \item $-9$
    \item 9
    \item 18
    \item 27
  \end{enumerate}

\end{enumerate}
\end{multicols}
