% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A hiker ascends $5$ mountains, each with an elevation of $-200$ feet below sea level. What is the total elevation change?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-1000$ feet
    \item $1000$ feet
    \item $0$ feet
    \item $-500$ feet
    \item $500$ feet
  \end{enumerate}

  \item \small A park ranger tracks a pack of wolves, observing their population grows by $3$ every $2$ months. If the pack starts with $17$ wolves, how many will there be after $8$ months?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $23$ wolves
    \item $25$ wolves
    \item $29$ wolves
    \item $31$ wolves
    \item $35$ wolves
  \end{enumerate}

  \item \small A gas tank is filled with $25$ gallons of fuel. If $3$ gallons are consumed every $50$ miles, how many miles can the vehicle travel on $-5$ gallons of fuel?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item The vehicle cannot travel
    \item $-80$ miles
    \item $-50$ miles
    \item $0$ miles
    \item $50$ miles
  \end{enumerate}

  \item \small What is the next number in the pattern: $2, 5, 8, 11, 14, ...$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $17$
    \item $20$
    \item $21$
    \item $23$
    \item $25$
  \end{enumerate}

  \item \small A group of friends plan a trip to a national park, and they drive a total distance of $240$ miles. If they drive at an average speed of $-8$ miles per hour, what is their total travel time?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item The trip is not possible
    \item $-30$ hours
    \item $-20$ hours
    \item $0$ hours
    \item $30$ hours
  \end{enumerate}

  \item \small What is the result of $(-3) \cdot (-5)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-15$
    \item $15$
    \item $-9$
    \item $9$
    \item $0$
  \end{enumerate}

  \item \small A camper observes that the temperature drops by $2$ degrees every hour. If the initial temperature is $-5$ degrees, what is the temperature after $4$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-13$ degrees
    \item $-9$ degrees
    \item $-5$ degrees
    \item $-1$ degree
    \item $1$ degree
  \end{enumerate}

  \item \small A hiker needs to ascend a mountain with an elevation change of $1200$ feet. If the hiker climbs $4$ hours a day at a rate of $-3$ feet per hour, how many days will it take to reach the top?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item The hiker will never reach the top
    \item $-100$ days
    \item $100$ days
    \item $50$ days
    \item $40$ days
  \end{enumerate}

\end{enumerate}
\end{multicols}
