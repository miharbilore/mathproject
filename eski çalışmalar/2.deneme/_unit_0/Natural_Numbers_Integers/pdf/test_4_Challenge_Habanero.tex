\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A recipe for making bread requires $3 \cdot 2$ cups of flour. How many cups of flour are needed in total?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 6
    \item 7
    \item 8
  \end{enumerate}

  \item \small A building has $15$ floors. If the architect wants to add $2$ more floors, what will be the new total number of floors?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 13
    \item 15
    \item 17
    \item 19
    \item 21
  \end{enumerate}

  \item \small Which of the following numbers is an even integer: $-4$, $-3$, $-2$, $-1$, $0$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-4$
    \item $-3$
    \item $-2$
    \item $-1$
    \item $0$
  \end{enumerate}

  \item \small A chef needs to package $18$ cookies into boxes that hold $3$ cookies each. How many boxes can the chef fill?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 6
    \item 7
    \item 8
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $x + 2 = 7$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3
    \item 5
    \item 7
    \item 9
    \item 11
  \end{enumerate}

  \item \small A contractor needs to lay $240$ meters of pipe. If the pipe is sold in $8$-meter lengths, how many pieces of pipe will the contractor need to buy?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 28
    \item 30
    \item 32
    \item 35
    \item 40
  \end{enumerate}

  \item \small Which of the following is a natural number: $-1$, $0$, $1$, $-2$, $-3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-1$
    \item $0$
    \item $1$
    \item $-2$
    \item $-3$
  \end{enumerate}

  \item \small A recipe calls for $2 \cdot 3$ teaspoons of sugar. If the recipe is repeated $2$ times, how many teaspoons of sugar are needed in total?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8
    \item 10
    \item 12
    \item 14
    \item 16
  \end{enumerate}

\end{enumerate}
