% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small In a game, a player's score can be represented as $x \cdot y$, where $x$ and $y$ are integers. If $x = -3$ and $y = 4$, what is the score?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item -12
    \item 12
    \item -3
    \item 4
    \item 0
  \end{enumerate}

  \item \small A coding error occurs every $5^{th}$ line of code. If a programmer writes $23$ lines of code, how many errors will occur?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 3
    \item 2
    \item 1
  \end{enumerate}

  \item \small The number of players in an online game follows the pattern: $2, 5, 8, 11, 14, ...$. What is the next number in the pattern?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 17
    \item 18
    \item 20
    \item 21
  \end{enumerate}

  \item \small In a scoreboard, a team's score is represented as $-2x + 5$, where $x$ is an integer. If the team's score is $-3$, what is the value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 4
    \item 5
    \item 6
  \end{enumerate}

  \item \small A computer program uses $3$ integers to generate a code: $x, y, z$. If $x = -2, y = 4, z = -1$, what is the value of $x + y + z$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1
    \item 3
    \item 0
    \item -1
    \item -3
  \end{enumerate}

  \item \small The number of bytes used by a game is given by the formula $2^n - 1$, where $n$ is an integer. If $n = 5$, how many bytes are used?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 31
    \item 32
    \item 33
    \item 63
    \item 65
  \end{enumerate}

  \item \small A programmer uses the pattern $1, 4, 7, 10, 13, ...$ to generate a sequence of numbers. What is the $6^{th}$ term in the sequence?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 16
    \item 17
    \item 19
    \item 20
  \end{enumerate}

  \item \small The score of a player in a game is represented as $x - 2y$, where $x$ and $y$ are integers. If $x = 5$ and $y = -3$, what is the score?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 11
    \item 9
    \item 7
    \item 5
    \item 3
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A computer program generates a sequence of numbers using the formula $2n + 1$, where $n$ is an integer. If the sequence starts with $3, 5, 7, ...$, find the $10^{th}$ term in the sequence. Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
