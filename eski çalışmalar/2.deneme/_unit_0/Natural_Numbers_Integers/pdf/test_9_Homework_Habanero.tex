% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Classify the number $-5$ as a:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Natural Number
    \item Whole Number
    \item Integer
    \item Rational Number
    \item Irrational Number
  \end{enumerate}

  \item \small A doctor prescribes $3$ pills every $4$ hours. How many pills will a patient take in $24$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$
    \item $18$
    \item $20$
    \item $24$
    \item $30$
  \end{enumerate}

  \item \small Identify the pattern: $2, 5, 8, 11, 14, ...$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Add $2$
    \item Add $3$
    \item Add $4$
    \item Add $5$
    \item Multiply by $2$
  \end{enumerate}

  \item \small Solve the equation $x + 2 = 7$ for $x$:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$
    \item $7$
    \item $9$
    \item $-5$
    \item $-7$
  \end{enumerate}

  \item \small A cell grows at a rate of $15%$ per hour. If there are initially $50$ cells, how many cells will there be after $2$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $57.5$
    \item $60$
    \item $65$
    \item $72.5$
    \item $75$
  \end{enumerate}

  \item \small Evaluate the expression $(-3) \cdot (-5)$:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $-15$
    \item $15$
    \item $-9$
    \item $9$
    \item $0$
  \end{enumerate}

  \item \small Determine if $27$ is an even or odd number:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Even
    \item Odd
    \item Both
    \item Neither
    \item Cannot be determined
  \end{enumerate}

  \item \small Solve the inequality $2x > 10$ for $x$:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 5$
    \item $x < 5$
    \item $x = 5$
    \item $x \geq 5$
    \item $x \leq 5$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A patient is taking a medication that has a $20%$ chance of side effects. If $50$ patients are taking the medication, how many are expected to experience side effects? Show your work and explain your reasoning. \vspace{2cm}

  \item \small A population of bacteria grows according to the equation $P(t) = 200 \cdot 2^{0.5t}$, where $t$ is time in hours. Find the population after $4$ hours and explain the significance of the growth rate. \vspace{2cm}

\end{enumerate}
