\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the variable in the expression $2x + 3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x$
    \item $3$
    \item $x$
    \item $2$
    \item $+$
  \end{enumerate}

  \item \small Identify the constant term in the expression $4x - 2$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4x$
    \item $-$
    \item $-2$
    \item $4$
    \item $x$
  \end{enumerate}

  \item \small What is the coefficient of $x$ in $5x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x$
    \item $5$
    \item $5x$
    \item $1$
    \item $-$
  \end{enumerate}

  \item \small Which symbol is the operator in the expression $7 	imes 2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $7$
    \item $2$
    \item $	imes$
    \item $=$
    \item $+$
  \end{enumerate}

  \item \small Identify the expression that has a variable: $2 + 3$ or $2x + 3$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2 + 3$
    \item $2x + 3$
    \item Both
    \item Neither
    \item None
  \end{enumerate}

  \item \small Which of the following is an example of a simple expression? $2x + 5$ or $2x + 5y$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x + 5$
    \item $2x + 5y$
    \item Both
    \item Neither
    \item None
  \end{enumerate}

  \item \small What operation is represented by the $+$ symbol in $3 + 4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Subtraction
    \item Addition
    \item Multiplication
    \item Division
    \item Exponentiation
  \end{enumerate}

  \item \small Which of the following expressions contains more than one operation? $2x$ or $2x + 3$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2x$
    \item $2x + 3$
    \item Both
    \item Neither
    \item None
  \end{enumerate}

\end{enumerate}
