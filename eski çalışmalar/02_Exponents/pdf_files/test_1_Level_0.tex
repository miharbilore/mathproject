% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is $2^3$ equal to?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6$
    \item $7$
    \item $8$
    \item $9$
    \item $10$
  \end{enumerate}

  \item \small What is $x^2$ when $x = 3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6$
    \item $7$
    \item $8$
    \item $9$
    \item $10$
  \end{enumerate}

  \item \small What is the value of $5^1$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $5$
    \item $10$
    \item $25$
    \item $50$
  \end{enumerate}

  \item \small What is $3^2$ equal to?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6$
    \item $7$
    \item $8$
    \item $9$
    \item $10$
  \end{enumerate}

  \item \small What is the meaning of $x^0$ when $x \neq 0$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $x$
    \item $-x$
    \item $\frac{1}{x}$
  \end{enumerate}

  \item \small What is $2^4$ equal to?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$
    \item $12$
    \item $16$
    \item $20$
    \item $24$
  \end{enumerate}

  \item \small What is $1^5$ equal to?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $5$
    \item $10$
    \item $15$
  \end{enumerate}

  \item \small What does the term $a^m$ represent?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $a$ multiplied by itself $m$ times
    \item $a$ divided by itself $m$ times
    \item $a$ added to itself $m$ times
    \item $a$ subtracted from itself $m$ times
    \item $a$ raised to the power of $m-1$
  \end{enumerate}

\end{enumerate}
\end{multicols}
