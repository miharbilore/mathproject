\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve for $x$: $2x + 5 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
  \end{enumerate}

  \item \small A patient is taking a medication that increases in dosage by $3$ milligrams each day. If the initial dosage is $10$ milligrams, what is the dosage after $4$ days?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10 + 3 \cdot 4 = 22$ milligrams
    \item $10 + 2 \cdot 4 = 18$ milligrams
    \item $10 + 4 \cdot 3 = 22$ milligrams
    \item $10 - 3 \cdot 4 = -2$ milligrams
    \item $10 - 2 \cdot 4 = 2$ milligrams
  \end{enumerate}

  \item \small Solve for $y$: $y - 2 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 5$
    \item $y = 6$
    \item $y = 9$
    \item $y = 10$
    \item $y = 11$
  \end{enumerate}

  \item \small The number of bacteria in a petri dish doubles every $2$ hours. If there are initially $100$ bacteria, how many will there be after $6$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $100 \cdot 2^3 = 800$ bacteria
    \item $100 \cdot 2^2 = 400$ bacteria
    \item $100 \cdot 2^1 = 200$ bacteria
    \item $100 \cdot 2^4 = 1600$ bacteria
    \item $100 \cdot 2^5 = 3200$ bacteria
  \end{enumerate}

  \item \small Solve for $z$: $z/4 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $z = 36$
    \item $z = 40$
    \item $z = 45$
    \item $z = 32$
    \item $z = 28$
  \end{enumerate}

  \item \small A medical researcher is studying the effect of a new drug on blood pressure. The researcher finds that for every $2$ milligrams of the drug, blood pressure decreases by $1$ point. If a patient's initial blood pressure is $140$ points, what dosage will lower their blood pressure to $120$ points?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2 \cdot (140 - 120) = 40$ milligrams
    \item $2 \cdot (140 - 120) = 20$ milligrams
    \item $1 \cdot (140 - 120) = 20$ milligrams
    \item $2 \cdot (120 - 140) = -40$ milligrams
    \item $1 \cdot (140 - 120) = 20$ milligrams
  \end{enumerate}

  \item \small Solve for $x$: $x + 2 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 5$
    \item $x = 7$
    \item $x = 11$
    \item $x = 13$
    \item $x = 15$
  \end{enumerate}

  \item \small A pharmaceutical company is producing a new medication. The cost of producing $x$ units of the medication is given by $C(x) = 2x + 1000$. What is the cost of producing $250$ units?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $C(250) = 2 \cdot 250 + 1000 = 1500$
    \item $C(250) = 2 \cdot 250 + 1000 = 2000$
    \item $C(250) = 2 \cdot 250 + 1000 = 2500$
    \item $C(250) = 2 \cdot 250 + 1000 = 3000$
    \item $C(250) = 2 \cdot 250 + 1000 = 15000$
  \end{enumerate}

\end{enumerate}
