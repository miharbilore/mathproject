% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve for $x$: $x + 2 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 5$
    \item $x = 9$
    \item $x = 3$
    \item $x = 1$
    \item $x = 6$
  \end{enumerate}

  \item \small Solve for $y$: $y - 4 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 7$
    \item $y = 15$
    \item $y = 9$
    \item $y = 12$
    \item $y = 8$
  \end{enumerate}

  \item \small A stock's price increased by $3. If its original price was $25, what is its new price?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $28$
    \item $30$
    \item $22$
    \item $32$
    \item $20$
  \end{enumerate}

  \item \small Solve for $z$: $2z = 18$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $z = 8$
    \item $z = 9$
    \item $z = 10$
    \item $z = 7$
    \item $z = 6$
  \end{enumerate}

  \item \small A savings account earns $5 in interest. If it had $120 initially, what is its new balance?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $120$
    \item $115$
    \item $125$
    \item $130$
    \item $100$
  \end{enumerate}

  \item \small Solve for $x$: $x/3 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 3$
    \item $x = 27$
    \item $x = 12$
    \item $x = 6$
    \item $x = 9$
  \end{enumerate}

  \item \small A tax rate of $8%$ is applied to a $250 purchase. How much tax is paid?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $25$
    \item $15$
    \item $10$
    \item $18$
  \end{enumerate}

  \item \small Solve for $y$: $y - 2 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 5$
    \item $y = 9$
    \item $y = 10$
    \item $y = 4$
    \item $y = 6$
  \end{enumerate}

\end{enumerate}
\end{multicols}
