% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A hiker's backpack weighs $4x + 5$ pounds. If the backpack weighs 21 pounds, what is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 4
    \item 5
    \item 6
  \end{enumerate}

  \item \small A car travels $250$ miles on $10$ gallons of gas. If $x$ gallons are used, how many miles are traveled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $25x$ miles
    \item $10x$ miles
    \item $20x$ miles
    \item $15x$ miles
    \item $5x$ miles
  \end{enumerate}

  \item \small A park ranger observes that a population of rabbits grows according to the equation $y = 2x + 5$. If $y = 11$, what is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 4
    \item 5
    \item 6
  \end{enumerate}

  \item \small Solve for $x$: $x + 2 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 7$
    \item $x = 11$
    \item $x = 5$
    \item $x = 3$
    \item $x = 1$
  \end{enumerate}

  \item \small A group of friends want to go on a $3$-day hike. Their food supply is given by $y = 3x - 2$. If they have $7$ days of food, how many days will they actually hike?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3 days
    \item 5 days
    \item 7 days
    \item 9 days
    \item 11 days
  \end{enumerate}

  \item \small A car's fuel consumption is given by $y = 2x + 1$. If $y = 5$, what is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1
    \item 2
    \item 3
    \item 4
    \item 5
  \end{enumerate}

  \item \small Solve for $x$: $4x - 2 = 2x + 6$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 4$
    \item $x = 6$
    \item $x = 8$
    \item $x = 10$
  \end{enumerate}

  \item \small A trail's length is given by $y = x + 3$. If the trail is $9$ miles long, what is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 7
    \item 8
    \item 9
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A group of hikers are planning a trip. Their bus will travel $x$ miles to the starting point, and then they will hike $2x + 5$ miles to the campsite. If the total distance is $35$ miles, find $x$ and show your work. \vspace{2cm}

  \item \small A car is traveling from City A to City B. The distance between the cities is $240$ miles. The car's speed is given by $v = 2x + 10$, where $v$ is in miles per hour and $x$ is in hours. If the trip takes $4$ hours, find $x$ and the car's average speed. \vspace{2cm}

\end{enumerate}
