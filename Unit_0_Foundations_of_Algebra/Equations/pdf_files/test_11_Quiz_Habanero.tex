% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve for $x$: $x + 5 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 6$
    \item $x = 7$
    \item $x = 8$
    \item $x = 9$
    \item $x = 10$
  \end{enumerate}

  \item \small A glacier is losing $3$ meters of ice per year. If it currently has $45$ meters of ice, how many years will it take for the glacier to lose $15$ meters of ice?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ years
    \item $3$ years
    \item $2$ years
    \item $1$ year
    \item $4$ years
  \end{enumerate}

  \item \small The rainfall in a desert region is increasing by $2$ inches per month. If the current rainfall is $10$ inches, how many months will it take for the rainfall to reach $16$ inches?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ months
    \item $3$ months
    \item $2$ months
    \item $6$ months
    \item $4$ months
  \end{enumerate}

  \item \small Solve for $y$: $2y - 4 = 12$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 6$
    \item $y = 7$
    \item $y = 8$
    \item $y = 9$
    \item $y = 10$
  \end{enumerate}

  \item \small The temperature in a greenhouse is rising by $1.5$ degrees Celsius per hour. If the current temperature is $20$ degrees Celsius, how many hours will it take for the temperature to reach $25$ degrees Celsius?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ hours
    \item $3$ hours
    \item $2$ hours
    \item $1.67$ hours
    \item $4$ hours
  \end{enumerate}

  \item \small Solve for $z$: $z/2 + 3 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $z = 6$
    \item $z = 7$
    \item $z = 8$
    \item $z = 9$
    \item $z = 10$
  \end{enumerate}

  \item \small A lake is being filled at a rate of $0.5$ meters per day. If the lake currently has $2$ meters of water and needs $4$ meters to be considered full, how many days will it take to fill the lake?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$ days
    \item $3$ days
    \item $4$ days
    \item $6$ days
    \item $8$ days
  \end{enumerate}

  \item \small Solve for $x$: $x - 2 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 7$
    \item $x = 10$
    \item $x = 11$
    \item $x = 12$
    \item $x = 13$
  \end{enumerate}

\end{enumerate}
