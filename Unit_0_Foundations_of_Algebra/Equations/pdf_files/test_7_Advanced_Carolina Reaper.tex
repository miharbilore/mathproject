% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve for $x$ in the equation $2x + 5 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
  \end{enumerate}

  \item \small A book costs $$15$ more than a pencil. If $x$ represents the cost of the pencil, the equation is $x + 15 = 25$. Solve for $x$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 10$
    \item $x = 12$
    \item $x = 15$
    \item $x = 18$
    \item $x = 20$
  \end{enumerate}

  \item \small Solve the equation $4x - 2 = 2x + 6$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 4$
    \item $x = 6$
    \item $x = 8$
    \item $x = 10$
  \end{enumerate}

  \item \small Tom has been saving money for a new bike and has $x$ dollars. He wants to buy a bike that costs $120. If he has already saved $$40$, the equation is $x + 40 = 120$. Solve for $x$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 60$
    \item $x = 70$
    \item $x = 80$
    \item $x = 90$
    \item $x = 100$
  \end{enumerate}

  \item \small Solve for $y$ in the equation $y - 3 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 4$
    \item $y = 5$
    \item $y = 10$
    \item $y = 11$
    \item $y = 12$
  \end{enumerate}

  \item \small A group of friends want to split a bill evenly. If the total bill is $$48$ and there are $x$ friends, the equation is $48 = 8x$. Solve for $x$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
    \item $x = 7$
    \item $x = 8$
  \end{enumerate}

  \item \small Solve the equation $2x + 2 = 5x - 3$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 1$
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
  \end{enumerate}

  \item \small Ashley has been practicing a new song on her guitar. If she practices for $x$ hours per day and wants to practice for a total of 12 hours, the equation is $3x = 12$. Solve for $x$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
  \end{enumerate}

\end{enumerate}
\end{multicols}
