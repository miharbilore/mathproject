    \item $w = 12$ pounds
    \item $w = 15$ pounds
    \item $w = 18$ pounds
    \item $w = 20$ pounds
    \item $w = 10$ pounds
  \end{enumerate}

  \item \small Solve for $y$: $y - 4 = 11$

    \item $y = 20$
    \item $y = 12$
    \item $y = 9$
  \end{enumerate}

  \item \small A math artifact from the Mayan civilization has $2x + 5 = 11$. What is $x$?

    \item $x = 2$
    \item $x = 3$
    \item $x = 1$

    \item $x = 6$

    \item $x = 5$
    \item $x = 8$
    \item $x = 4$
