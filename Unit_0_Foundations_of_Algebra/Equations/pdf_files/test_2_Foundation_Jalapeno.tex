    \item $x = 1$
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
  \end{enumerate}

  \item \small Solve for $y$: $2y - 4 = 8$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]

    \item $y = 4$
    \item $y = 5$
    \item $y = 6$
    \item $y = 7$
  \end{enumerate}

  \item \small A lab experiment requires a temperature of $2x + 5$ degrees. If the temperature is 11 degrees, what is the value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 1$
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
  \end{enumerate}

  \item \small Solve for $z$: $z - 2 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $z = 7$
    \item $z = 8$
    \item $z = 9$
    \item $z = 10$
    \item $z = 11$
  \end{enumerate}

  \item \small The height of a rocket is $5x - 3$ feet after $x$ seconds. If the height is 22 feet, how many seconds have passed?

    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
    \item $x = 7$
  \end{enumerate}

  \item \small Solve for $w$: $w + 2 = 6$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]

    \item $w = 3$
    \item $w = 4$
    \item $w = 5$
    \item $w = 6$
  \end{enumerate}

    \item $x = 4$
