% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve for $x$: $2x + 5 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
  \end{enumerate}

  \item \small A gamer has $120$ points. After a level, their points are given by $3x - 20$. If they now have $200$ points, what is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 20$
    \item $x = 30$
    \item $x = 40$
    \item $x = 50$
    \item $x = 60$
  \end{enumerate}

  \item \small A coding error causes a program to use $2x + 10$ GB of data. If the program should use $5x - 20$ GB, what is $x$ if $2x + 10 = 5x - 20$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 5$
    \item $x = 10$
    \item $x = 15$
    \item $x = 20$
    \item $x = 25$
  \end{enumerate}

  \item \small In a game, $3x - 2 = 5x + 1$. What is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = -1$
    \item $x = -0.5$
    \item $x = 0$
    \item $x = 0.5$
    \item $x = 1$
  \end{enumerate}

  \item \small Solve for $y$: $y - 3 = 2(y + 2)$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = -1$
    \item $y = 0$
    \item $y = 1$
    \item $y = 2$
    \item $y = 3$
  \end{enumerate}

  \item \small A scoreboard shows $x + 2 = 7$. What is $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
    \item $x = 7$
  \end{enumerate}

  \item \small Solve for $z$: $z/2 + 2 = 6$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $z = 4$
    \item $z = 6$
    \item $z = 8$
    \item $z = 10$
    \item $z = 12$
  \end{enumerate}

  \item \small A gaming PC uses $x$ GB of RAM. If $x - 4 = 7$, how much RAM does it use?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 3$
    \item $x = 7$
    \item $x = 11$
    \item $x = 15$
    \item $x = 20$
  \end{enumerate}

\end{enumerate}
\end{multicols}
