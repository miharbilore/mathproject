\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A package is traveling from City $A$ to City $B$ at a rate of $x$ miles per hour. If the distance between the two cities is $3x + 10$ miles and the package takes $x + 2$ hours to arrive, solve for $x$ and check your solution for accuracy. Show all work. \vspace{2cm}

  \item \small A logistics company has $5x + 2$ packages to deliver in $x$ hours. If each package requires $2x - 1$ minutes to deliver, how many packages can be delivered per hour? Solve for $x$ and check your solution for accuracy. Show all work. \vspace{2cm}

\end{enumerate}
