% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A shipping company has $3 \cdot x + 2$ trucks on the road, where $x$ is the number of delivery routes. If $x = 4$, how many trucks are on the road?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 10
    \item 12
    \item 14
    \item 16
    \item 18
  \end{enumerate}

  \item \small Evaluate the expression $2^3 + 12 \div 4 - 5$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3
    \item 5
    \item 7
    \item 9
    \item 11
  \end{enumerate}

  \item \small A traffic flow model is given by the expression $x^2 - 4x + 4$. Simplify the expression.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^2 - 4x$
    \item $x^2 + 4$
    \item $x^2 - 4x + 4$
    \item $x^2 - 2x$
    \item $x^2 + 2x$
  \end{enumerate}

  \item \small The time it takes to deliver a package is given by $t = 2x + 5$, where $t$ is the time in hours and $x$ is the distance in miles. If $x = 10$, what is the delivery time?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15 hours
    \item 20 hours
    \item 25 hours
    \item 30 hours
    \item 35 hours
  \end{enumerate}

  \item \small Evaluate the expression $(3 + 2) \cdot (8 - 3)$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20
    \item 25
    \item 30
    \item 35
    \item 40
  \end{enumerate}

  \item \small A logistics company has a cost function given by $C = 2x^2 + 5x + 10$, where $C$ is the cost and $x$ is the number of shipments. If $x = 3$, what is the total cost?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $37
    \item $43
    \item $49
    \item $55
    \item $61
  \end{enumerate}

  \item \small Simplify the expression $3(2x + 1) - 2(x - 3)$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4x + 7$
    \item $5x - 2$
    \item $6x + 1$
    \item $7x - 3$
    \item $8x + 2$
  \end{enumerate}

  \item \small Evaluate the expression $(9 - 3) \div (2 + 1)$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1
    \item 2
    \item 3
    \item 4
    \item 5
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A transportation company has a route that takes $t = x^2 - 4x + 5$ hours to complete, where $t$ is the time and $x$ is the distance in miles. If the company wants to deliver packages to two locations, one at $x = 5$ miles and the other at $x = 10$ miles, what is the total time taken for both deliveries? Show your work and calculate the total time. \vspace{2cm}

  \item \small A shipping container has a volume given by $V = 2x^2 + 5x - 3$, where $V$ is the volume in cubic feet and $x$ is the length in feet. If the container is 8 feet long, what is the volume of the container? Show your work and calculate the volume. \vspace{2cm}

\end{enumerate}
