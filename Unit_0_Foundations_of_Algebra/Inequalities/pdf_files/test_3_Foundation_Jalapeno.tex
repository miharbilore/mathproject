% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve the inequality $2x + 5 > 11$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 3$
    \item $x < 3$
    \item $x > -3$
    \item $x < -3$
    \item $x = 3$
  \end{enumerate}

  \item \small A basketball player scores $x$ points per game. If she scores more than 15 points per game, which inequality represents this situation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x < 15$
    \item $x > 15$
    \item $x = 15$
    \item $x leq 15$
    \item $x geq 15$
  \end{enumerate}

  \item \small Graph the solution to $x - 2 < 7$ on a number line.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-infty, 9)$
    \item $(-infty, 5)$
    \item $(-infty, 10)$
    \item $[9, infty)$
    \item $[5, infty)$
  \end{enumerate}

  \item \small A cyclist rides $x$ miles in $y$ hours. If she rides more than 20 miles in 2 hours, which inequality represents this situation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x < 20$
    \item $x > 20$
    \item $y < 2$
    \item $y > 2$
    \item $x = 20$
  \end{enumerate}

  \item \small Solve the inequality $x/4 + 2 > 6$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 16$
    \item $x < 16$
    \item $x > 8$
    \item $x < 8$
    \item $x = 16$
  \end{enumerate}

  \item \small Write the solution to $x + 1 > 9$ in interval notation.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(8, infty)$
    \item $(-infty, 8)$
    \item $[8, infty)$
    \item $(-infty, 8]$
    \item $(-8, infty)$
  \end{enumerate}

  \item \small A runner runs $x$ kilometers in a race. If she runs more than 5 kilometers in less than 30 minutes, which inequality represents this situation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x < 5$
    \item $x > 5$
    \item $x = 5$
    \item $x leq 5$
    \item $x geq 5$
  \end{enumerate}

  \item \small Solve the inequality $3x - 2 < 14$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x < 16/3$
    \item $x > 16/3$
    \item $x < 5$
    \item $x > 5$
    \item $x = 16/3$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A football team scores $x$ points per game. If they score more than 10 points per game, and the score is reduced by 2 points per game due to penalties, write the inequality that represents the new score and graph the solution on a number line. Require a minimum of 3 lines to answer. \vspace{2cm}

If she performs more than 3 flips and each flip has a value of $y$ points, write the inequality that represents the total score and solve for $x$. Assume $y$ is a constant and $x$ and $y$ are positive integers. Require a minimum of 4 lines to answer. \vspace{2cm}

\end{enumerate}
