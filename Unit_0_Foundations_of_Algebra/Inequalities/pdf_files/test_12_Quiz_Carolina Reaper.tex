% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A cargo ship travels $120$ miles in $t$ hours. If the speed is at least $20$ mph, what inequality represents this situation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $t \geq 6$
    \item $t \leq 6$
    \item $t > 6$
    \item $t < 6$
    \item $t = 6$
  \end{enumerate}

  \item \small The cost $C$ of shipping a package is given by $C = 5x + 10$, where $x$ is the weight in pounds. If the cost is at most $$25$, what is the inequality for $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x \leq 3$
    \item $x \geq 3$
    \item $x < 3$
    \item $x > 3$
    \item $x = 3$
  \end{enumerate}

  \item \small A traffic light has a red phase that lasts for $t$ seconds. If $t$ is between $20$ and $40$ seconds, the inequality for $t$ is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20 < t < 40$
    \item $20 \leq t \leq 40$
    \item $t > 20$
    \item $t < 40$
    \item $t = 20$
  \end{enumerate}

  \item \small The number of boxes $n$ that can be delivered by a truck is given by $n = 10t - 5$, where $t$ is the time in hours. If the number of boxes is at least $30$, what inequality represents this situation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $t \geq 3.5$
    \item $t \leq 3.5$
    \item $t > 3.5$
    \item $t < 3.5$
    \item $t = 3.5$
  \end{enumerate}

  \item \small A company offers a discount on shipping for orders over $50$. If $x$ represents the cost of the order, the inequality for orders that qualify for the discount is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 50$
    \item $x \geq 50$
    \item $x < 50$
    \item $x \leq 50$
    \item $x = 50$
  \end{enumerate}

  \item \small A delivery person can carry at most $25$ packages. If $x$ represents the number of packages, the inequality for $x$ is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x \leq 25$
    \item $x \geq 25$
    \item $x < 25$
    \item $x > 25$
    \item $x = 25$
  \end{enumerate}

  \item \small The time $t$ it takes for a package to be delivered is given by $t = \frac{x}{10} + 2$, where $x$ is the distance in miles. If the delivery time is at most $5$ hours, what inequality represents this situation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x \leq 30$
    \item $x \geq 30$
    \item $x < 30$
    \item $x > 30$
    \item $x = 30$
  \end{enumerate}

  \item \small A truck can travel at most $400$ miles in a day. If $x$ represents the distance traveled, the inequality for $x$ is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x \leq 400$
    \item $x \geq 400$
    \item $x < 400$
    \item $x > 400$
    \item $x = 400$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A package delivery company guarantees that packages will be delivered within $t$ hours, where $t = \frac{x}{5} + 1$, and $x$ is the distance in miles. Graph the inequality $t \leq 4$ on a number line and write the solution in interval notation. \vspace{2cm}

  \item \small A cargo ship travels from port $A$ to port $B$ at a speed of $s$ miles per hour. The time $t$ it takes for the ship to travel from $A$ to $B$ is given by $t = \frac{240}{s}$. If the time $t$ is between $4$ and $12$ hours, write an inequality for $s$ and graph the solution on a number line. \vspace{2cm}

\end{enumerate}
