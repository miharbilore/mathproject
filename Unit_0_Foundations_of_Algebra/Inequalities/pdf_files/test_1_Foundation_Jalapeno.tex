\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve the inequality $2x + 5 > 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 3$
    \item $x < 3$
    \item $x > -3$
    \item $x < -3$
    \item $x = 3$
  \end{enumerate}

  \item \small Graph the solution to $x - 2 < 5$ on a number line
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-\infty, 7)$
    \item $(-\infty, 3)$
    \item $[3, \infty)$
    \item $(-\infty, -3)$
    \item $[7, \infty)$
  \end{enumerate}

  \item \small Write the solution to $x + 1 > -4$ in interval notation
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-3, \infty)$
    \item $(-\infty, -3)$
    \item $(-4, \infty)$
    \item $(-\infty, -4)$
    \item $(-5, \infty)$
  \end{enumerate}

  \item \small A stock's price is $25, and it increases by $2 every day. How many days will it take for the price to be greater than $35?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 6 days
    \item 5 days
    \item 4 days
    \item 10 days
    \item 8 days
  \end{enumerate}

  \item \small If $3x - 2 > 7$, then what is the value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 3$
    \item $x < 3$
    \item $x = 3$
    \item $x > 2$
    \item $x < 2$
  \end{enumerate}

  \item \small Solve the inequality $5x - 3 < 2x + 2$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x < 1$
    \item $x > 1$
    \item $x > -1$
    \item $x < -1$
    \item $x = 1$
  \end{enumerate}

  \item \small Graph the solution to $x + 2 > 1$ on a number line
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-\infty, -1)$
    \item $(-1, \infty)$
    \item $(-\infty, 1)$
    \item $(-\infty, -3)$
    \item $[1, \infty)$
  \end{enumerate}

  \item \small Write the solution to $2x + 1 < 9$ in interval notation
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-\infty, 4)$
    \item $(-\infty, -4)$
    \item $(-4, \infty)$
    \item $(-\infty, 3)$
    \item $(-3, \infty)$
  \end{enumerate}

\end{enumerate}
