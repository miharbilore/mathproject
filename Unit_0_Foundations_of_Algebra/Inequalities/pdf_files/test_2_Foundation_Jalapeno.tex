% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve the inequality $2x + 5 > 11$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 3$
    \item $x < 3$
    \item $x geq 3$
    \item $x leq 3$
    \item $x = 3$
  \end{enumerate}

  \item \small Graph the solution to the inequality $x - 2 < 7$ on a number line.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-infty, 9)$
    \item $(-infty, 5)$
    \item $(-infty, 3)$
    \item $(5, infty)$
    \item $(9, infty)$
  \end{enumerate}

  \item \small Write the solution to the inequality $4x > 28$ in interval notation.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-infty, 7)$
    \item $(7, infty)$
    \item $(-infty, -7)$
    \item $(-7, infty)$
    \item $[7, infty)$
  \end{enumerate}

  \item \small A space probe is traveling at a speed of $5x - 2$ miles per hour. If it needs to travel faster than $23$ miles per hour, what is the minimum value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 5$
    \item $x < 5$
    \item $x geq 5$
    \item $x leq 5$
    \item $x = 5$
  \end{enumerate}

  \item \small Solve the inequality $rac{3x}{4} < 12$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x < 16$
    \item $x > 16$
    \item $x geq 16$
    \item $x leq 16$
    \item $x = 16$
  \end{enumerate}

  \item \small A scientist is conducting an experiment with a temperature range of $-10$ to $30$ degrees Celsius. If the temperature is greater than $10$ degrees Celsius, what is the solution in interval notation?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-10, 10)$
    \item $(10, 30)$
    \item $(-10, 30)$
    \item $(-infty, 10)$
    \item $(10, infty)$
  \end{enumerate}

  \item \small Solve the inequality $x + 2 > 9$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 7$
    \item $x < 7$
    \item $x geq 7$
    \item $x leq 7$
    \item $x = 7$
  \end{enumerate}

  \item \small Graph the solution to the inequality $3x - 1 < 14$ on a number line.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-infty, 5)$
    \item $(-infty, 3)$
    \item $(-infty, 2)$
    \item $(5, infty)$
    \item $(3, infty)$
  \end{enumerate}

\end{enumerate}
\end{multicols}
