% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve the inequality $2x - 5 > 11$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 8$
    \item $x < 8$
    \item $x > 4$
    \item $x < 4$
    \item $x = 8$
  \end{enumerate}

  \item \small Graph the solution to $x + 2 \leq 7$ on a number line.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-\infty, 5]$
    \item $[-\infty, 5]$
    \item $[5, \infty)$
    \item $(-\infty, 5)$
    \item $[5, \infty]$
  \end{enumerate}

  \item \small Write the solution to $x - 3 < 9$ in interval notation.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $(-\infty, 12)$
    \item $(-\infty, 6)$
    \item $[6, \infty)$
    \item $(-\infty, 6]$
    \item $[6, \infty]$
  \end{enumerate}

  \item \small If the temperature is $15^\circ$C and rises by $2^\circ$C every hour, when will it reach $25^\circ$C?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5 hours
    \item 6 hours
    \item 4 hours
    \item 3 hours
    \item 2 hours
  \end{enumerate}

  \item \small Solve the inequality $4x + 2 > 3x + 5$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x > 3$
    \item $x < 3$
    \item $x > 1$
    \item $x < 1$
    \item $x = 3$
  \end{enumerate}

  \item \small If the rainfall in a forest is $x$ mm and the maximum allowed is $200$ mm, write an inequality to represent $x$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x \leq 200$
    \item $x \geq 200$
    \item $x < 200$
    \item $x > 200$
    \item $x = 200$
  \end{enumerate}

  \item \small Graph the solution to $2x - 1 \geq 5$ on a number line.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $[3, \infty)$
    \item $(-\infty, 3)$
    \item $(-\infty, 3]$
    \item $[3, \infty]$
    \item $(-\infty, 3)$
  \end{enumerate}

  \item \small If the temperature drops by $5^\circ$C every hour and is currently $20^\circ$C, when will it reach $0^\circ$C?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5 hours
    \item 4 hours
    \item 3 hours
    \item 2 hours
    \item 1 hour
  \end{enumerate}

\end{enumerate}
\end{multicols}
