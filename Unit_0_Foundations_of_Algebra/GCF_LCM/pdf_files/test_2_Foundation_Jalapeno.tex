\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the GCF of $12$ and $18$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 6
    \item 9
    \item 12
  \end{enumerate}

  \item \small A space lab has $15$ packets of oxygen and $20$ packets of food. What is the LCM of $15$ and $20$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 30
    \item 60
    \item 75
    \item 100
    \item 120
  \end{enumerate}

  \item \small What is the GCF of $24$ and $30$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 6
    \item 12
    \item 24
  \end{enumerate}

  \item \small A physics experiment requires $8$ repetitions of a task. If each repetition takes $12$ minutes, what is the LCM of $8$ and $12$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 12
    \item 16
    \item 24
    \item 32
    \item 48
  \end{enumerate}

  \item \small What is the GCF of $9$ and $15$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1
    \item 3
    \item 5
    \item 9
    \item 15
  \end{enumerate}

  \item \small A rocket has $18$ communication satellites and $24$ navigation satellites. What is the GCF of $18$ and $24$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 6
    \item 9
    \item 12
  \end{enumerate}

  \item \small What is the LCM of $9$ and $12$ using multiples?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 18
    \item 24
    \item 27
    \item 36
    \item 48
  \end{enumerate}

  \item \small A space station has $20$ modules with $15$ astronauts in each module. What is the LCM of $20$ and $15$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 30
    \item 45
    \item 60
    \item 75

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A lab experiment requires $12$ containers of a chemical and $18$ containers of another chemical. If each container of the first chemical weighs $3$ kg and each container of the second chemical weighs $2$ kg, what is the total weight of the chemicals if the GCF of $12$ and $18$ is used to optimize storage? \vspace{2cm}

  \item \small A space mission has $15$ crew members who need to be divided into teams of $3$ or $5$ for training. If the LCM of $3$ and $5$ is used to determine the number of teams, how many teams can be formed and what is the total number of crew members in these teams? \vspace{2cm}

\end{enumerate}
