% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A hiker is packing energy bars for a trip. If she needs $12$ bars for the first day and $15$ bars for the second day, what is the least common multiple (LCM) of $12$ and $15$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $30$
    \item $36$
    \item $60$
    \item $72$
    \item $90$
  \end{enumerate}

  \item \small A fuel tank can hold $24$ gallons of gasoline. If a car uses $4$ gallons to travel $10$ miles, how many gallons will it use to travel $30$ miles? What is the greatest common factor (GCF) of $24$ and $4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $6$
    \item $8$
  \end{enumerate}

  \item \small In a forest, the population of rabbits is $48$ and the population of deer is $18$. What is the GCF of $48$ and $18$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A park ranger needs to distribute food to monkeys. If she has $20$ bags of food and each bag contains $6$ servings, how many servings are there in total? What is the LCM of $20$ and $6$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $30$
    \item $60$
    \item $120$
    \item $180$
  \end{enumerate}

  \item \small A group of friends want to go on a hike that is $45$ miles long. If they can hike $5$ miles per day, how many days will it take them to complete the hike? What is the GCF of $45$ and $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $3$
    \item $5$
    \item $9$
    \item $15$
  \end{enumerate}

  \item \small A car travels $24$ miles per gallon of gasoline. If the gas tank can hold $16$ gallons, how many miles can the car travel on a full tank? What is the LCM of $24$ and $16$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $48$
    \item $64$
    \item $96$
    \item $128$
    \item $192$
  \end{enumerate}

  \item \small A wildlife sanctuary has $50$ animals in one enclosure and $25$ animals in another. What is the GCF of $50$ and $25$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $5$
    \item $10$
    \item $25$
    \item $50$
  \end{enumerate}

  \item \small A hiker needs to climb $36$ flights of stairs to reach the top of a mountain. If she can climb $4$ flights per hour, how many hours will it take her to reach the top? What is the LCM of $36$ and $4$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$
    \item $18$
    \item $36$
    \item $48$
    \item $72$
  \end{enumerate}

\end{enumerate}
\end{multicols}
