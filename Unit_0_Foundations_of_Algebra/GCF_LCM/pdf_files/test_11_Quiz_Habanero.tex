% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the GCF of $12$ and $18$ using factors?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small Find the LCM of $9$ and $15$ using multiples.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $30$
    \item $45$
    \item $60$
    \item $90$
  \end{enumerate}

  \item \small A storm is forecasted to bring $24$ inches of rain to a coastal town. If the town's water reservoir can hold $48$ inches of water, what is the GCF of the total rainfall and the reservoir's capacity?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $8$
    \item $12$
  \end{enumerate}

  \item \small The temperature in a city changes by $8$ degrees every hour. If it also changes by $12$ degrees every two hours, what is the LCM of these two temperature changes?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$
    \item $12$
    \item $16$
    \item $24$
    \item $32$
  \end{enumerate}

  \item \small A group of scientists are studying the population growth of a certain ecosystem. They find that the population of one species increases by $15$ every $3$ days, and another species increases by $20$ every $5$ days. What is the LCM of the time intervals for these population increases?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3$
    \item $5$
    \item $10$
    \item $15$
    \item $15$
  \end{enumerate}

  \item \small What is the GCF of $24$, $36$, and $48$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $4$
    \item $12$
  \end{enumerate}

  \item \small The rainfall in a region follows a cycle of $18$ days of rain and $12$ days of dry weather. What is the LCM of the days of rain and dry weather?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$
    \item $18$
    \item $30$
    \item $36$
    \item $54$
  \end{enumerate}

  \item \small A team of environmentalists are monitoring the water level of a lake. The water level rises by $9$ inches every $2$ hours and falls by $6$ inches every $3$ hours. What is the GCF of the time intervals for these changes?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $9$
  \end{enumerate}

\end{enumerate}
\end{multicols}
