  \item \small What is the GCF of $12$ and $18$ using factors?

  \end{enumerate}

  \item \small A gamer has $24$ hours of gameplay and wants to divide it into segments with the same duration. If the segments must be a factor of $24$, what are the possible segment lengths? What is the LCM of the possible segment lengths $4$, $6$, $8$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $24$
    \item $48$
    \item $96$
    \item $120$
    \item $144$
  \end{enumerate}

  \item \small What is the GCF of $20$ and $30$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$
    \item $5$
    \item $10$
    \item $2 \cdot 5$
    \item $2 \cdot 2 \cdot 5$
  \end{enumerate}

  \item \small A coding app has $15$ and $25$ minute levels. What is the LCM of $15$ and $25$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $25$
    \item $75$
    \item $100$
    \item $150$
  \end{enumerate}

  \item \small What is the GCF of $9$ and $12$ using factors?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $4$
    \item $6$
  \end{enumerate}

  \item \small A data plan has $300$ MB and $600$ MB limits. What is the LCM of $300$ and $600$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $300$
    \item $600$
    \item $900$
    \item $1200$
    \item $1800$
  \end{enumerate}

  \item \small A scoreboard has $8$ and $12$ point rewards. What is the GCF of $8$ and $12$ using prime factorization?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$
    \item $2 \cdot 2$
    \item $2 \cdot 2 \cdot 2$
    \item $2 \cdot 2 \cdot 2 \cdot 2$
    \item $4$
  \end{enumerate}

  \item \small A game developer wants to create levels with $18$, $24$, and $30$ minute durations. What is the LCM of $18$, $24$, and $30$?
