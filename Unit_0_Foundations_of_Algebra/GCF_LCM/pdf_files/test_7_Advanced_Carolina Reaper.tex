% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Find the GCF of $12$ and $15$ using factors.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $3$
    \item $5$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A bookshelf has $18$ shelves, and each shelf can hold $12$ books. What is the LCM of $18$ and $12$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $36$
    \item $48$
    \item $54$
    \item $60$
    \item $72$
  \end{enumerate}

  \item \small Find the LCM of $8$ and $10$ using prime factorization.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $30$
    \item $40$
    \item $50$
    \item $60$
  \end{enumerate}

  \item \small A music store sells guitar strings in packs of $6$ or $8$. What is the GCF of $6$ and $8$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $6$
    \item $8$
  \end{enumerate}

  \item \small A group of friends want to go on a road trip that is $240$ miles long. If they drive $15$ miles per hour, how many hours will it take them to complete the trip, assuming they drive for $8$ hours a day? Use GCF and LCM to solve.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$ hours
    \item $16$ hours
    \item $18$ hours
    \item $20$ hours
    \item $24$ hours
  \end{enumerate}

  \item \small Find the GCF of $24$ and $30$ using prime factorization.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A bakery sells bread in loaves of $12$ or $15$. What is the LCM of $12$ and $15$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $36$
    \item $48$
    \item $60$
    \item $72$
    \item $120$
  \end{enumerate}

  \item \small A water tank can hold $120$ gallons of water. If $8$ gallons of water are added to the tank every hour, how many hours will it take to fill the tank? Use GCF and LCM to solve.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10$ hours
    \item $12$ hours
    \item $15$ hours
    \item $18$ hours
    \item $20$ hours
  \end{enumerate}

\end{enumerate}
\end{multicols}
