    \item $6$
    \item $12$
    \item $18$
    \item $24$
    \item $36$
  \end{enumerate}

  \item \small What is the GCF of $48$ and $60$ using prime factorization?

    \item $2$
    \item $3$
    \item $4$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A doctor prescribes a patient to take $15$ pills in the morning and $20$ pills in the evening. What is the LCM of $15$ and $20$ to determine the smallest number of days the patient needs to take the pills?

    \item $30$
    \item $45$
    \item $60$
    \item $75$
    \item $90$
  \end{enumerate}

  \item \small What is the GCF of $27$ and $45$ using factors?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $3$
    \item $5$
    \item $9$
    \item $15$
  \end{enumerate}

  \item \small A biologist is studying the growth of cells. The cells grow in cycles of $8$ and $12$ hours. What is the LCM of $8$ and $12$ to determine the time it takes for the cells to complete one full cycle?

    \item $16$
    \item $20$
    \item $24$

  \end{enumerate}

  \item \small What is the GCF of $50$ and $70$ using prime factorization?
