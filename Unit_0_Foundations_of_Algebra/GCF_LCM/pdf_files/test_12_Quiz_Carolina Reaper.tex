% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A shipping company has $12$ and $15$ minute delivery intervals. What is the $GCF(12, 15)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $3$
    \item $5$
    \item $6$
    \item $9$
  \end{enumerate}

  \item \small A traffic light has $8$ and $12$ second intervals. What is the $LCM(8, 12)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$
    \item $12$
    \item $24$
    \item $32$
    \item $48$
  \end{enumerate}

  \item \small A truck travels $18$ miles in $3$ hours and another truck travels $24$ miles in $4$ hours. What is the $GCF(18, 24)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A train has $20$ and $30$ minute stops. What is the $LCM(20, 30)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $20$
    \item $30$
    \item $40$
    \item $60$
    \item $120$
  \end{enumerate}

  \item \small A delivery service has packages of $9$ and $12$ items. What is the $GCF(9, 12)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $4$
    \item $6$
  \end{enumerate}

  \item \small A bus travels $15$ miles in $3$ hours and another bus travels $25$ miles in $5$ hours. What is the $LCM(15, 25)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $25$
    \item $50$
    \item $75$
    \item $125$
  \end{enumerate}

  \item \small A taxi company has $8$ and $10$ minute pickup intervals. What is the $GCF(8, 10)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $6$
    \item $8$
  \end{enumerate}

  \item \small A shipping crate has $6$ and $9$ foot dimensions. What is the $LCM(6, 9)$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6$
    \item $9$
    \item $12$
    \item $18$
    \item $36$
  \end{enumerate}

\end{enumerate}
