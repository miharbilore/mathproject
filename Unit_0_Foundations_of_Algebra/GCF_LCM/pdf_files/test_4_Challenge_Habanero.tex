\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A recipe for $3 \cdot 4$ pancakes requires $2 \cdot 3$ eggs. What is the GCF of the number of pancakes and eggs?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 6
    \item 12
    \item 1
  \end{enumerate}

  \item \small A construction site has $12$ and $15$ meter long wooden planks. What is the LCM of these lengths?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 30
    \item 60
    \item 120
    \item 180
    \item 240
  \end{enumerate}

  \item \small A chef needs to package $48$ cookies into boxes that hold $8$ or $12$ cookies. What is the GCF of the number of cookies per box?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 4
    \item 8
    \item 12
    \item 1
  \end{enumerate}

  \item \small A builder is constructing a fence with $6$ and $9$ foot long sections. What is the LCM of these lengths?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 18
    \item 27
    \item 36
    \item 54
    \item 108
  \end{enumerate}

  \item \small A recipe for $2 \cdot 3$ cakes requires $5 \cdot 2$ cups of flour. What is the LCM of the number of cakes and cups of flour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 30
    \item 60
    \item 120
    \item 180
    \item 240
  \end{enumerate}

  \item \small A construction crew needs to mix $20$ and $30$ liters of cement. What is the GCF of the amounts of cement?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 5
    \item 10
    \item 20
    \item 1
  \end{enumerate}

  \item \small A chef is preparing a dish that requires $9$ and $12$ cups of ingredients. What is the LCM of the number of cups?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 36
    \item 54
    \item 72
    \item 108
    \item 144
  \end{enumerate}

  \item \small A builder is designing a building with $8$ and $12$ foot high walls. What is the GCF of the wall heights?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 4
    \item 8
    \item 12
    \item 1
  \end{enumerate}

\end{enumerate}
