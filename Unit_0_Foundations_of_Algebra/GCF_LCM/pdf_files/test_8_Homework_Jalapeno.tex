% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small What is the GCF of $12$ and $18$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small The LCM of $9$ and $15$ is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $45$
    \item $90$
    \item $135$
    \item $225$
  \end{enumerate}

  \item \small In ancient Egypt, the Pyramids of Giza were built with stones weighing $12$, $18$, or $24$ tons. What is the GCF of these weights?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small The prime factorization of $42$ is $2 \cdot 3 \cdot 7$. What is the LCM of $42$ and $63$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $42$
    \item $63$
    \item $126$
    \item $252$
    \item $315$
  \end{enumerate}

  \item \small A Mayan calendar has $15$ days in a week and $20$ weeks in a month. What is the LCM of $15$ and $20$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $20$
    \item $60$
    \item $75$
    \item $100$
  \end{enumerate}

  \item \small The GCF of $24$ and $30$ is
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small A Greek architect is designing a temple with columns spaced $8$ meters apart and a hallway $12$ meters wide. What is the GCF of $8$ and $12$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $8$
    \item $12$
  \end{enumerate}

  \item \small The prime factorization of $48$ is $2^4 \cdot 3$. What is the LCM of $48$ and $75$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $48$
    \item $75$
    \item $300$
    \item $600$
    \item $1200$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small In the construction of the Great Wall of China, stones were placed $10$ meters apart. If a watchtower was built every $25$ meters, what is the LCM of $10$ and $25$? Show your work and explain your reasoning. \vspace{2cm}

  \item \small An ancient Babylonian farmer had $18$ baskets of dates and $24$ baskets of figs. If he wanted to pack the baskets in rows with the same number of baskets in each row, what is the GCF of $18$ and $24$? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
