% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Find the GCF of $12$ and $20$ using prime factorization.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$
    \item $4$
    \item $6$
    \item $8$
    \item $12$
  \end{enumerate}

  \item \small A song has a rhythm pattern that repeats every $15$ seconds. If another song has a pattern that repeats every $25$ seconds, what is the LCM of their pattern lengths?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $75$
    \item $100$
    \item $125$
    \item $150$
    \item $175$
  \end{enumerate}

  \item \small Find the LCM of $9$ and $12$ using multiples.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $18$
    \item $24$
    \item $27$
    \item $36$
    \item $48$
  \end{enumerate}

  \item \small A painter uses a $12$-inch brush and a $16$-inch brush to paint a canvas. What is the GCF of the brush lengths in inches?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $4$
    \item $6$
    \item $8$
  \end{enumerate}

  \item \small A music producer wants to mix two songs with frequencies of $12$ Hz and $15$ Hz. What is the LCM of their frequencies?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $24$
    \item $30$
    \item $36$
    \item $60$
    \item $72$
  \end{enumerate}

  \item \small A canvas has a width of $18$ inches and a length of $24$ inches. What is the GCF of the dimensions in inches?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1$
    \item $2$
    \item $3$
    \item $6$
    \item $12$
  \end{enumerate}

  \item \small Find the GCF of $20$ and $30$ using factors.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$
    \item $10$
    \item $15$
    \item $20$
    \item $25$
  \end{enumerate}

  \item \small A drummer has two rhythms with frequencies of $8$ Hz and $12$ Hz. What is the LCM of their frequencies?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $16$
    \item $20$
    \item $24$
    \item $32$
    \item $40$
  \end{enumerate}

\end{enumerate}
\end{multicols}
