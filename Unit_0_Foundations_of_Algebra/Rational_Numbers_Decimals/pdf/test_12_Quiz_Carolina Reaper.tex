% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A shipping company has $3 \frac{1}{4}$ hours to deliver a package. What is this time in decimal form?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 3.1
    \item 3.25
    \item 3.5
    \item 3.75
    \item 4.0
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $2 \cdot x = 1 \frac{3}{4}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.75
    \item 0.875
    \item 1.0
    \item 1.25
    \item 1.5
  \end{enumerate}

  \item \small A traffic flow study found that the average commute time was $45 \frac{1}{2}$ minutes. What percent of an hour is this?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 72.5%
    \item 75%
    \item 76.25%
    \item 80%
    \item 82.5%
  \end{enumerate}

  \item \small A delivery truck travels $2 \frac{3}{8}$ miles in $\frac{1}{4}$ hour. What is its average speed in miles per hour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8
    \item 9.5
    \item 10
    \item 12
    \item 16
  \end{enumerate}

  \item \small What is the decimal equivalent of $\frac{5}{8}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.5
    \item 0.6
    \item 0.625
    \item 0.75
    \item 0.8
  \end{enumerate}

  \item \small A logistics company has a $3:2$ ratio of trucks to drivers. If there are $4 \frac{1}{2}$ drivers, how many trucks are there?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 6.75
    \item 7
    \item 8
  \end{enumerate}

  \item \small A package delivery service guarantees $98 \frac{3}{4}$% on-time delivery. What is the decimal equivalent of this percentage?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0.983
    \item 0.984
    \item 0.985
    \item 0.9875
    \item 0.988
  \end{enumerate}

  \item \small What is the value of $x$ in the equation $\frac{2}{3}x = 1 \frac{1}{8}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 1.5
    \item 1.75
    \item 2.0
    \item 2.25
    \item 2.5
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A traffic engineer wants to represent the average commute times on a number line, with values ranging from $0$ to $2$ hours. If the average commute time is $1 \frac{1}{4}$ hours, where would it be placed on the number line? Show your work and explain your answer. \vspace{2cm}

  \item \small A shipping company has a $2:5$ ratio of overnight to standard deliveries. If $4 \frac{3}{8}$ standard deliveries were made, how many overnight deliveries were made? Show your work and explain your answer. \vspace{2cm}

\end{enumerate}
