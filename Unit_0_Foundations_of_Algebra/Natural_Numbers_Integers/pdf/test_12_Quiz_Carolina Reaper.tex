% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A shipping company has $5 \cdot 3$ trucks. How many trucks does the company have?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 13
    \item 15
    \item 20
    \item 25
    \item 30
  \end{enumerate}

  \item \small The traffic flow rate is increasing by $2$ cars per minute. If there are currently $10$ cars on the road, how many will there be in $4$ minutes?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 12
    \item 14
    \item 16
    \item 18
    \item 22
  \end{enumerate}

  \item \small A delivery driver is $-3$ hours behind schedule. How many hours will it take to deliver $6$ packages if each package takes $2$ hours to deliver?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 9
    \item 12
    \item 15
    \item 18
    \item 21
  \end{enumerate}

  \item \small Find the next even number in the pattern: $2, 6, 10, 14, ...$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 16
    \item 18
    \item 20
    \item 22
    \item 24
  \end{enumerate}

  \item \small A logistics company has $12$ crates to transport. If $4$ crates are already loaded, what is the value of $x$ in the equation $12 - 4 = x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 6
    \item 8
    \item 10
    \item 12
    \item 14
  \end{enumerate}

  \item \small Determine the value of $x$ in the equation $x + 5 = 11$ if $x$ is an integer.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 7
    \item 8
    \item 9
  \end{enumerate}

  \item \small A transportation company has a total of $25$ vehicles, including $15$ trucks and $10$ cars. If $x$ represents the number of trucks, which equation represents the total number of vehicles?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x + 10 = 25$
    \item $x - 10 = 25$
    \item $x + 15 = 25$
    \item $x - 15 = 25$
    \item $x = 25 - 10$
  \end{enumerate}

  \item \small A pattern of odd numbers starts with $3, 7, 11, 15, ...$. What is the next odd number in the pattern?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 17
    \item 19
    \item 21
    \item 23
    \item 25
  \end{enumerate}

\end{enumerate}
\end{multicols}

\end{enumerate}
