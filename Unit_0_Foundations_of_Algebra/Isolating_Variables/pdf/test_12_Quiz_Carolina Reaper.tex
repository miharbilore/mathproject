% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Solve for $x$ in the equation $2x + 5 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 2$
    \item $x = 3$
    \item $x = 4$
    \item $x = 5$
    \item $x = 6$
  \end{enumerate}

  \item \small A truck is traveling from City A to City B at an average speed of $x$ miles per hour. If the distance between the two cities is $240$ miles and the travel time is $4$ hours, what is the value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 50$
    \item $x = 60$
    \item $x = 70$
    \item $x = 80$
    \item $x = 90$
  \end{enumerate}

  \item \small Solve for $y$ in the equation $y - 3 = 7$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 4$
    \item $y = 5$
    \item $y = 10$
    \item $y = 11$
    \item $y = 12$
  \end{enumerate}

  \item \small A logistics company has a fleet of $x$ trucks, and each truck can carry $15$ crates. If the company needs to transport $315$ crates, what is the value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 15$
    \item $x = 20$
    \item $x = 21$
    \item $x = 25$
    \item $x = 30$
  \end{enumerate}

  \item \small Solve for $z$ in the equation $z + 2 = 9$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $z = 5$
    \item $z = 6$
    \item $z = 7$
    \item $z = 10$
    \item $z = 11$
  \end{enumerate}

  \item \small A ship is traveling from Port A to Port B at an average speed of $x$ knots. If the distance between the two ports is $180$ nautical miles and the travel time is $6$ hours, what is the value of $x$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x = 20$
    \item $x = 25$
    \item $x = 30$
    \item $x = 35$
    \item $x = 40$
  \end{enumerate}

  \item \small Solve for $w$ in the equation $w - 2 = 11$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $w = 9$
    \item $w = 10$
    \item $w = 12$
    \item $w = 13$
    \item $w = 14$
  \end{enumerate}

  \item \small A traffic flow model is given by the equation $x + 2y = 12$, where $x$ is the number of cars and $y$ is the number of trucks. Solve for $y$ when $x = 4$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $y = 2$
    \item $y = 3$
    \item $y = 4$
    \item $y = 5$
    \item $y = 6$
  \end{enumerate}

\end{enumerate}
\end{multicols}
