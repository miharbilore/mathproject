% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A cargo ship can transport $480$ containers of goods. If each container holds $12$ crates, how many crates can the ship transport in total, and what is the remainder when divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5760$ R $0$
    \item $5760$ R $3$
    \item $5760$ R $6$
    \item $5760$ R $9$
    \item $5761$ R $1$
  \end{enumerate}

  \item \small A truck driver needs to deliver $945$ packages to a warehouse. If he delivers them in batches of $15$, how many batches can he make, and what is the remainder?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $63$ R $0$
    \item $63$ R $2$
    \item $63$ R $5$
    \item $63$ R $10$
    \item $62$ R $15$
  \end{enumerate}

  \item \small A traffic light cycle lasts $120$ seconds. How many times will it cycle in $18$ minutes, using $2$ as a divisor?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $9$ R $0$
    \item $10$ R $0$
    \item $9$ R $2$
    \item $8$ R $4$
    \item $18$ R $0$
  \end{enumerate}

  \item \small A logistics company has $270$ parcels to ship. If they are shipped in boxes of $9$, how many boxes can be filled, and what is the remainder?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $29$ R $3$
    \item $30$ R $0$
    \item $30$ R $3$
    \item $29$ R $6$
    \item $30$ R $6$
  \end{enumerate}

  \item \small A delivery service needs to divide $216$ packages among $12$ vans. What is the quotient and remainder when $216$ is divided by $12$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $18$ R $0$
    \item $17$ R $6$
    \item $18$ R $4$
    \item $18$ R $6$
    \item $19$ R $6$
  \end{enumerate}

  \item \small A bus travels $630$ miles in one day. If it travels the same route $5$ times a week, how many miles does it travel per week using $5$ as a divisor?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3150$
    \item $3150$ R $0$
    \item $3151$ R $1$
    \item $3120$ R $10$
    \item $3152$
  \end{enumerate}

  \item \small A shipping company has $504$ boxes of cargo. If they are packed in containers that hold $8$ boxes each, how many containers can be filled, and what is the remainder?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $63$ R $0$
    \item $63$ R $1$
    \item $63$ R $4$
    \item $62$ R $8$
    \item $63$ R $8$
  \end{enumerate}

  \item \small A train travels $810$ miles from one station to another. If it makes $5$ stops along the way, and each stop takes the same amount of time, what is the remainder when $810$ is divided by $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $2$
    \item $3$
    \item $5$
  \end{enumerate}

\end{enumerate}
\end{multicols}
