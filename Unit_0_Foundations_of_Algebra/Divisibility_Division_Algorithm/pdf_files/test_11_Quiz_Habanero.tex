% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A temperature sensor in a weather station measures temperatures in $\degree C$. If the sensor readings for 15 consecutive days are added, the sum is $225$. What is the average temperature reading if the sum is divisible by $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $15$
    \item $15.5$
    \item $16$
    \item $16.5$
    \item $17$
  \end{enumerate}

  \item \small What is the remainder when $27$ is divided by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $2$
    \item $3$
    \item $6$
  \end{enumerate}

  \item \small A water treatment plant can process $480$ gallons of water per hour. If $16$ hours are worked, how many gallons of water are treated in total, and is the total divisible by $10$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $7680$, yes
    \item $7680$, no
    \item $7681$, yes
    \item $7681$, no
    \item $7682$, yes
  \end{enumerate}

  \item \small What is the remainder when $43$ is divided by $5$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $2$
    \item $3$
    \item $8$
  \end{enumerate}

  \item \small If $18$ is divided by $3$, what is the quotient and is it divisible by $2$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $6$, yes
    \item $6$, no
    \item $7$, yes
    \item $7$, no
    \item $8$, yes
  \end{enumerate}

  \item \small A researcher tracks the rainfall in an ecosystem and finds that the total rainfall for $9$ consecutive days is $279$ mm. Is the average rainfall divisible by $3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item yes
    \item no
    \item not enough information
    \item maybe
    \item divisible by $9$
  \end{enumerate}

  \item \small What is the remainder when $75$ is divided by $10$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $0$
    \item $1$
    \item $2$
    \item $5$
    \item $8$
  \end{enumerate}

  \item \small A temperature reading of $35$ $\degree F$ is equivalent to how many degrees Celsius, and is the result divisible by $9$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1.67$, yes
    \item $1.67$, no
    \item $2$, yes
    \item $2.01$, no
    \item $2.22$, yes
  \end{enumerate}

\end{enumerate}
\end{multicols}
