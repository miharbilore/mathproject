\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A music artist wants to arrange $48$ songs into boxes of $8$ songs each. How many boxes can be filled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 6 with 0 songs leftover
    \item 6 with 0 songs leftover, because $48 \div 8 = 6$
    \item 8
  \end{enumerate}

  \item \small A beat has a rhythm pattern that repeats every $15$ seconds. If a song is $5$ minutes long, how many times will the rhythm pattern repeat?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 15
    \item 20
    \item 30
    \item 32
    \item 40
  \end{enumerate}

  \item \small If a painting is $18$ inches by $24$ inches, what is the greatest common divisor of its dimensions?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 2
    \item 3
    \item 6
    \item 8
    \item 12
  \end{enumerate}

  \item \small To create a harmonic frequency, a musician needs to divide a $480$ Hz signal into $16$ equal parts. What is the resulting frequency of each part?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20 Hz
    \item 30 Hz
    \item 30 Hz, because $480 \div 16 = 30$
    \item 40 Hz
    \item 60 Hz
  \end{enumerate}

  \item \small A photographer has $54$ prints to arrange in an exhibit. If she wants to display them in rows of $9$, how many rows can she make?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 6 with 0 prints leftover
    \item 6 with 0 prints leftover, because $54 \div 9 = 6$
    \item 8
  \end{enumerate}

  \item \small A composer needs to divide $36$ instrumental parts among $4$ musicians. How many parts will each musician get?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 8
    \item 9
    \item 9 with 0 parts leftover
    \item 9 with 0 parts leftover, because $36 \div 4 = 9$
    \item 12
  \end{enumerate}

  \item \small A sculptor has $50$ blocks of stone to arrange into a pattern that repeats every $10$ blocks. How many times will the pattern repeat?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 5 with 0 blocks leftover
    \item 5 with 0 blocks leftover, because $50 \div 10 = 5$
    \item 8
  \end{enumerate}

  \item \small An artist has $24$ colors to divide into boxes that hold $4$ colors each. How many boxes can be filled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 6
    \item 6 with 0 colors leftover
    \item 8
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A music producer wants to create a $3$-minute song with a beat that repeats every $12$ seconds. If the song needs to have $5$-second pauses between beats, how many beats can fit in the song? Show your calculation and express your answer as a division result with a remainder, if any. \vspace{2cm}

  \item \small A painter needs to divide a $60$-inch canvas into sections that are $5$ inches long. If the painter wants to add a $2$-inch border around each section, what is the total length of the canvas that will be used for the borders? Calculate the number of sections that can fit on the canvas, then find the total border length. \vspace{2cm}

\end{enumerate}
