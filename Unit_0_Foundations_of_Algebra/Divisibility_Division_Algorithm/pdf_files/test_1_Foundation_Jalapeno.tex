% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small Which of the following numbers is divisible by $2$ and $3$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 7
    \item 10
    \item 12
    \item 15
    \item 21
  \end{enumerate}

  \item \small If a stock market investor has $45$ shares to sell, how many complete sets of $9$ shares can be sold?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 4
    \item 5
    \item 6
    \item 7
    \item 8
  \end{enumerate}

  \item \small Determine if $75$ is divisible by $5$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Yes
    \item No
    \item Maybe
    \item Cannot determine
    \item Insufficient information
  \end{enumerate}

  \item \small A savings account has $240$ dollars. If this amount is divided into sets of $10$, how many complete sets can be made?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 20
    \item 22
    \item 24
    \item 26
    \item 28
  \end{enumerate}

  \item \small If $18$ is divided by $3$, what is the remainder?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 0
    \item 1
    \item 2
    \item 3
    \item 4
  \end{enumerate}

  \item \small How many times does $9$ go into $63$ with a remainder of $0$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item 5
    \item 6
    \item 7
    \item 8
    \item 9
  \end{enumerate}

  \item \small Determine the divisibility of $48$ by $2$ and $8$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item Divisible by $2$ only
    \item Divisible by $8$ only
    \item Divisible by both $2$ and $8$
    \item Not divisible by either
    \item Cannot determine
  \end{enumerate}

  \item \small A tax amount of $150$ dollars is to be divided among $5$ people. How much will each person pay?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $25$
    \item $30$
    \item $35$
    \item $40$
    \item $45$
  \end{enumerate}

\end{enumerate}
\end{multicols}
