% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A weather station records $15$ inches of rainfall in $3$ hours. What is the rate of rainfall per hour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3$ inches/hour
    \item $5$ inches/hour
    \item $6$ inches/hour
    \item $8$ inches/hour
    \item $10$ inches/hour
  \end{enumerate}

  \item \small If $y$ varies directly with $x$, and $y = 12$ when $x = 4$, find the value of $y$ when $x = 6$.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $12$
    \item $16$
    \item $18$
    \item $20$
    \item $24$
  \end{enumerate}

  \item \small A ship travels $240$ miles in $4$ hours. What is its average speed?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $40$ miles/hour
    \item $50$ miles/hour
    \item $60$ miles/hour
    \item $70$ miles/hour
    \item $80$ miles/hour
  \end{enumerate}

  \item \small The temperature in a city decreases by $2^circ$C for every $1000$ feet of elevation. If the temperature at sea level is $25^circ$C, find the temperature at an elevation of $3000$ feet.
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $19^circ$C
    \item $21^circ$C
    \item $23^circ$C
    \item $25^circ$C
    \item $27^circ$C
  \end{enumerate}

  \item \small A water tank can be filled by a pump at a rate of $2$ cubic meters per minute. How many minutes will it take to fill a tank that holds $30$ cubic meters of water?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10$ minutes
    \item $12$ minutes
    \item $15$ minutes
    \item $18$ minutes
    \item $20$ minutes
  \end{enumerate}

  \item \small A group of friends want to share some candy equally. If they have $48$ pieces of candy and there are $8$ friends, how many pieces will each friend get?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4$ pieces
    \item $5$ pieces
    \item $6$ pieces
    \item $7$ pieces
    \item $8$ pieces
  \end{enumerate}

  \item \small A cyclist travels $25$ miles in $2$ hours. What is the cyclist's average speed in miles per hour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10$ miles/hour
    \item $12$ miles/hour
    \item $12.5$ miles/hour
    \item $15$ miles/hour
    \item $20$ miles/hour
  \end{enumerate}

  \item \small The cost of a ticket to a concert is directly proportional to the number of tickets purchased. If $2$ tickets cost $$20$, how much will $5$ tickets cost?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $$25$
    \item $$30$
    \item $$40$
    \item $$45$
    \item $$50$
  \end{enumerate}

\end{enumerate}
\end{multicols}

  \item \small A company is planning to build a new wind farm. The cost of the wind farm is directly proportional to the number of wind turbines. If the cost of $3$ wind turbines is $$900,000$, find the cost of $7$ wind turbines. Show your work and explain your reasoning. \vspace{2cm}

  \item \small A city's water reservoir can hold $1,200,000$ gallons of water. Due to a drought, the reservoir is being filled at a rate of $5,000$ gallons per hour. How many hours will it take to fill the reservoir? If the city's population is growing at a rate of $2%$ per year, and the current population is $100,000$, how many people will there be in $5$ years? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
