% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A truck is loaded with $2/3$ cement and $1/3$ sand. If $300$ kg of sand is added, what fraction of the mixture is now sand?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1/3$
    \item $1/2$
    \item $2/5$
    \item $3/7$
    \item $1/4$
  \end{enumerate}

  \item \small A shipping company has $480$ boxes of fragile items and $720$ boxes of non-fragile items. What is the ratio of fragile to non-fragile items?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2:3$
    \item $1:2$
    \item $3:4$
    \item $1:1$
    \item $4:5$
  \end{enumerate}

  \item \small A train has $30$ cars, each with $20%$ of its capacity filled with fuel. If $5$ more cars are added, each with $30%$ capacity filled, what percentage of the total capacity is now filled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $22%$
    \item $24%$
    \item $26%$
    \item $28%$
    \item $25%$
  \end{enumerate}

  \item \small A cargo ship travels $360$ km in $6$ hours with a $20%$ cargo load. How far will it travel in $4$ hours with a $30%$ cargo load, assuming the speed remains constant?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $200$ km
    \item $220$ km
    \item $240$ km
    \item $260$ km
    \item $280$ km
  \end{enumerate}

  \item \small A traffic flow model estimates that $30%$ of the cars on the road are traveling at $60$ km/h. If $20%$ are traveling at $80$ km/h, what percentage of the cars are traveling at speeds other than $60$ km/h or $80$ km/h?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $40%$
    \item $50%$
    \item $60%$
    \item $50%$
    \item $10%$
  \end{enumerate}

  \item \small A company has two types of delivery trucks, $A$ and $B$. Truck $A$ travels at $50$ km/h and truck $B$ travels at $75$ km/h. If $60%$ of the deliveries are made by truck $A$ and $40%$ by truck $B$, what is the average speed of all deliveries?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $57.5$ km/h
    \item $60$ km/h
    \item $62.5$ km/h
    \item $65$ km/h
    \item $67.5$ km/h
  \end{enumerate}

  \item \small A logistics company has a $25%$ margin on its shipping services. If it reduces its prices by $15%$, what is the new margin as a percentage of the original price?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $17.5%$
    \item $20%$
    \item $22.5%$
    \item $12.5%$
    \item $10%$
  \end{enumerate}

  \item \small A transportation company has two routes, $X$ and $Y$, with travel times of $2$ hours and $3$ hours respectively. If $40%$ of the passengers use route $X$ and $60%$ use route $Y$, what is the average travel time for all passengers?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2.4$ hours
    \item $2.6$ hours
    \item $2.8$ hours
    \item $3.0$ hours
    \item $3.2$ hours
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A shipping company has two types of containers, small and large. The small containers have a $20%$ chance of being damaged during shipping, while the large containers have a $10%$ chance. If $40%$ of the containers are small and $60%$ are large, what is the overall probability that a container will be damaged during shipping? Show your work and calculate the answer. \vspace{2cm}

  \item \small A train has $20$ cars, each with a $25%$ chance of being filled with cargo. If $5$ more cars are added, each with a $30%$ chance of being filled, what is the probability that exactly $12$ cars are filled with cargo? Use the binomial probability formula to calculate the answer. Show your work and provide the final probability. \vspace{2cm}

\end{enumerate}
