\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small A research team measures $2.5$ inches of rainfall in a forest and $1.8$ inches in a nearby meadow. What is the ratio of forest rainfall to meadow rainfall?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $1.35:1$
    \item $1.39:1$
    \item $1.45:1$
    \item $1.51:1$
    \item $1.58:1$
  \end{enumerate}

  \item \small If $15$ gallons of water are used to irrigate $30$ square feet of a garden, how many gallons are needed for $45$ square feet?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10$ gallons
    \item $15$ gallons
    \item $22.5$ gallons
    \item $25$ gallons
    \item $30$ gallons
  \end{enumerate}

  \item \small In an ecosystem, the ratio of predator to prey is $1:6$. If there are $36$ prey, how many predators are there?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$
    \item $6$
    \item $8$
    \item $10$
    \item $12$
  \end{enumerate}

  \item \small A water tank can hold $2400$ gallons of water. If $300$ gallons of water are already in the tank, what fraction of the tank is filled?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $rac{1}{8}$
    \item $rac{1}{6}$
    \item $rac{1}{5}$
    \item $rac{1}{4}$
    \item $rac{1}{3}$
  \end{enumerate}

  \item \small A temperature increase of $4^circ$F causes a $2$-inch rise in sea level. If the temperature increases by $12^circ$F, what is the rise in sea level?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ inches
    \item $6$ inches
    \item $8$ inches
    \item $10$ inches
    \item $12$ inches
  \end{enumerate}

  \item \small A park has a ratio of trees to benches of $12:1$. If there are $48$ trees, how many benches are there?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2$
    \item $3$
    \item $4$
    \item $5$
    \item $6$
  \end{enumerate}

  \item \small A cyclist travels $25$ miles in $2$ hours. If she maintains the same rate, how many miles will she travel in $4.5$ hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $50$ miles
    \item $56.25$ miles
    \item $62.5$ miles
    \item $67.5$ miles
    \item $70$ miles
  \end{enumerate}

  \item \small The ratio of sunny days to rainy days in a month is $5:3$. If there are $20$ rainy days, how many sunny days are there?

    \item $25$
    \item $30$
    \item $31$
    \item $32$
    \item $33$
