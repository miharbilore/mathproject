% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small If the temperature of a city is expected to increase by a factor of $2^3$ over the next 5 years, and it is currently $15^{circ}$C, what will the temperature be in 5 years?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $30^{circ}$C
    \item $40^{circ}$C
    \item $60^{circ}$C
    \item $80^{circ}$C
    \item $120^{circ}$C
  \end{enumerate}

  \item \small A certain type of plant can grow at a rate of $3^2$ times its original size when exposed to optimal rainfall conditions. If the plant is initially 5cm tall, how tall will it be after optimal rainfall?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5$ cm
    \item $15$ cm
    \item $30$ cm
    \item $45$ cm
    \item $60$ cm
  \end{enumerate}

  \item \small Simplify the expression $x^2 \cdot x^4$
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $x^6$
    \item $x^8$
    \item $x^{10}$
    \item $x^{12}$
    \item $x^{14}$
  \end{enumerate}

  \item \small A researcher measures the growth of a particular species of ecosystem, finding that its population grows by a factor of $(2^2)^3$ per year. What is the annual growth factor?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2^6$
    \item $2^8$
    \item $2^{10}$
    \item $2^{12}$
    \item $2^{14}$
  \end{enumerate}

  \item \small The volume of a cloud can be calculated using the formula $V = l^3$, where $l$ is the length of the cloud in kilometers. If the length is $2^2$ km, what is the volume?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $8$ km$^3$
    \item $16$ km$^3$
    \item $32$ km$^3$
    \item $64$ km$^3$
    \item $128$ km$^3$
  \end{enumerate}

  \item \small The rate of rainfall in a certain region can be modeled by the equation $R = 3^t$, where $R$ is the rate of rainfall and $t$ is time in hours. What is the rate of rainfall after 2 hours?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3$ mm/h
    \item $9$ mm/h
    \item $18$ mm/h
    \item $27$ mm/h
    \item $81$ mm/h
  \end{enumerate}

  \item \small A group of scientists study the effects of climate change, finding that the level of greenhouse gases in the atmosphere increases by a factor of $2^3$ every decade. What is the increase after 3 decades?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $2^6$
    \item $2^9$
    \item $2^{12}$
    \item $2^{15}$
    \item $2^{18}$
  \end{enumerate}

  \item \small The amount of solar radiation that hits the Earth's surface can be calculated using the formula $S = 4^t$, where $S$ is the amount of solar radiation and $t$ is time in hours. What is the amount of solar radiation after 1 hour?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4$ J/m$^2$
    \item $8$ J/m$^2$
    \item $16$ J/m$^2$
    \item $32$ J/m$^2$
    \item $64$ J/m$^2$
  \end{enumerate}

\end{enumerate}
\end{multicols}

\textbf{\small Open Ended Questions (FRQ)}
\begin{enumerate}[label=\textbf{Q\arabic*.}, start=9, itemsep=5pt, topsep=0pt]

  \item \small A certain type of ecosystem is sensitive to temperature changes, with its population decreasing by a factor of $3^2$ for every $2^{circ}$C increase in temperature. If the temperature is expected to increase by $2^3$ degrees over the next 10 years, and the current population is 1000, what will the population be in 10 years? Show your work and explain your reasoning. \vspace{2cm}

  \item \small The level of oxygen in a lake can be modeled by the equation $O = 2^{t-2}$, where $O$ is the level of oxygen and $t$ is time in months. If the current level of oxygen is 16, what will the level be after 5 months? Show your work and explain your reasoning. \vspace{2cm}

\end{enumerate}
