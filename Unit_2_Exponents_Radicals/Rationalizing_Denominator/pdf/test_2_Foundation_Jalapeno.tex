% MCQ Section
\begin{multicols}{2}
\begin{enumerate}[label=\textbf{Q\arabic*.}, leftmargin=*, itemsep=6pt, parsep=0pt]

  \item \small To rationalize the denominator of $\frac{1}{\sqrt{3}}$, we multiply by:
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{\sqrt{3}}{\sqrt{3}}$
    \item $\frac{2}{2}$
    \item $\frac{\sqrt{2}}{\sqrt{2}}$
    \item $\frac{1}{\sqrt{2}}$
    \item $\frac{\sqrt{6}}{\sqrt{6}}$
  \end{enumerate}

  \item \small What is the rationalized form of $\frac{2}{\sqrt{5}}$?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{2\sqrt{5}}{5}$
    \item $\frac{2}{5}$
    \item $\frac{\sqrt{5}}{2}$
    \item $\frac{2\sqrt{2}}{5}$
    \item $\frac{5}{2\sqrt{5}}$
  \end{enumerate}

  \item \small A spacecraft is traveling at $\frac{2}{\sqrt{3}}$ times the speed of light. What is the rationalized form of this speed?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{2\sqrt{3}}{3}$
    \item $\frac{2}{3}$
    \item $\frac{\sqrt{3}}{2}$
    \item $\frac{2\sqrt{2}}{3}$
    \item $\frac{3}{2\sqrt{3}}$
  \end{enumerate}

  \item \small The mass of a planet is $\frac{1}{\sqrt{2}}$ times the mass of Earth. What is the rationalized form of this mass?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{\sqrt{2}}{2}$
    \item $\frac{1}{2}$
    \item $\frac{2}{\sqrt{2}}$
    \item $\frac{\sqrt{2}}{\sqrt{2}}$
    \item $\frac{1}{\sqrt{2}}$
  \end{enumerate}

  \item \small The time it takes for a spaceship to travel to Mars is $\frac{3}{\sqrt{6}}$ years. What is the rationalized form of this time?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{\sqrt{6}}{2}$
    \item $\frac{3\sqrt{6}}{6}$
    \item $\frac{\sqrt{3}}{2}$
    \item $\frac{3}{2\sqrt{2}}$
    \item $\frac{\sqrt{2}}{3}$
  \end{enumerate}

  \item \small The distance between two planets is $\frac{2}{\sqrt{8}}$ light-years. What is the rationalized form of this distance?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{\sqrt{2}}{2}$
    \item $\frac{2\sqrt{2}}{4}$
    \item $\frac{\sqrt{8}}{2}$
    \item $\frac{2}{2\sqrt{2}}$
    \item $\frac{2\sqrt{8}}{8}$
  \end{enumerate}

  \item \small A lab experiment involves a chemical reaction with a rate of $\frac{1}{\sqrt{5}}$ moles per second. What is the rationalized form of this rate?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{\sqrt{5}}{5}$
    \item $\frac{1}{5}$
    \item $\frac{\sqrt{2}}{5}$
    \item $\frac{1}{\sqrt{2}}$
    \item $\frac{5}{\sqrt{5}}$
  \end{enumerate}

  \item \small The energy released by a star is $\frac{2}{\sqrt{3}}$ times the energy released by the Sun. What is the rationalized form of this energy?
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $\frac{2\sqrt{3}}{3}$
    \item $\frac{2}{3}$
    \item $\frac{\sqrt{3}}{2}$
    \item $\frac{2\sqrt{2}}{3}$
    \item $\frac{3}{2\sqrt{3}}$
  \end{enumerate}

\end{enumerate}
\end{multicols}
