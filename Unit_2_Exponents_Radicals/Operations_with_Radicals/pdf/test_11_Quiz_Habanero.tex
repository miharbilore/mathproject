  \item \small Simplify $\sqrt{48} + \sqrt{18}$

    \item $2\sqrt{6} + 3\sqrt{2}$
    \item $6\sqrt{3} + 2\sqrt{2}$
  \end{enumerate}

  \item \small The temperature in a greenhouse is $\sqrt{80}$ degrees higher than outside. Simplify this expression
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]

    \item $2\sqrt{10}$
    \item $8\sqrt{2}$
    \item $9\sqrt{3}$
    \item $6\sqrt{5}$
  \end{enumerate}

  \item \small $\sqrt{50} \cdot \sqrt{2} = $
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $10$
    \item $5\sqrt{2}$
    \item $10\sqrt{2}$
    \item $2\sqrt{25}$
    \item $25$
  \end{enumerate}

  \item \small A rainstorm causes the water level in a river to rise by $\sqrt{32}$ meters. Simplify this expression
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $4\sqrt{2}$
    \item $2\sqrt{8}$
    \item $8\sqrt{2}$
    \item $6\sqrt{3}$
    \item $3\sqrt{5}$
  \end{enumerate}

  \item \small $\frac{\sqrt{90}}{\sqrt{10}} = $
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $3$
    \item $\sqrt{3}$
    \item $\sqrt{9}$
    \item $3\sqrt{2}$
    \item $\sqrt{10}$
  \end{enumerate}

  \item \small $\sqrt{75} - \sqrt{48} = $
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]
    \item $5\sqrt{3} - 4\sqrt{3}$
    \item $3\sqrt{5} - 2\sqrt{6}$
    \item $4\sqrt{2} - 3\sqrt{2}$
    \item $5\sqrt{2} - 2\sqrt{5}$
    \item $2\sqrt{6} - 3\sqrt{3}$
  \end{enumerate}

  \item \small $\sqrt{20} \cdot \sqrt{5} = $
  \begin{enumerate}[label=(\Alph*), leftmargin=0.6cm, nosep, itemsep=2pt]

    \item $2\sqrt{10}$
    \item $10$
    \item $20\sqrt{2}$

  \end{enumerate}
